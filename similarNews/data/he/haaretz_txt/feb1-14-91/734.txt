היחסים בין ואשינגטון לממלכה ההאשמית בירדן הגיעו בימים האחרונים לשפל שכמותו 
לא היה מעולם . המשבר העמוק ממחיש היטב לממשל בוש , הרבה לפני שוך הסופה 
במדבר , שאין לו ברירה אלא לשכך את הרוחות הרעות שירדן מפיחה בעולם הערבי . 
המלך האשים ביום רביעי שעבר את הקואליציה שבהנהגת ארה"ב בניהול " מלחמה 
נגד כל המוסלמים ולא רק נגד עיראק , בניסיון לכונן " הגמוניה זרה " במזרח - 
התיכון ובניסיון " להרוס את המדינה - האחות , עיראק " . דברי הבלע המתסיסים 
הללו היו מרים כלענה לממשל המנסה בכל תוקף להכחיש את הטענות הללו ולבטל את 
הרושם , ההולך ומתגבש ברחוב הערבי , שכאילו מדובר במלחמת המערב נגד האיסלם 
או נגד המזרח הערבי . בממשל הלכה והתגבשה ההכרה , מאז ימי המלחמה הראשונים , 
כי ככל שיהיה ניתן לעכב את הגעש בעולם הערבי , כך יוכלו מדינות הברית להרוויח 
זמן בהפצצה האווירית שלהן כנגד עיראק ולחסוך דם במלחמה יבשתית . ככל שיגעש 
העולם הערבי , יגבר הלחץ על הממשל מצד מדינות ערביות בקואליציה , לזרז 
את המלחמה ולהביאה לסיום מהיר , גם במחיר של אבידות כבדות . נאום המלך " מסבך 
את העניינים " , אמר בתגובה הנשיא בוש למחרת השידור החי של נאום חוסיין ברשת 
סי - אן - אן . התחושה היתה שהמלך הגזים : " אם המלך רצה רק להפיס את 
דעתם של הקיצונים מבית , הוא לא היה צריך להשתמש במלים כל - כך קשות " , 
אמר פקיד בכיר במשרד החוץ של ארה"ב . הדעה הרווחת בקרב מעצבי המדיניות במשרד 
החוץ האמריקאי היא כי חוסיין לא אמר את מה שאמר רק בגלל הלחץ מהרחוב , אלא 
גם משום שהוא מאבד את אמונו בוואשינגטון . המלך הולך ומשתכנע , כך סבורים 
רבים בממשל , שאמריקה באמת רוקמת " מזימה אימפריאליסטית " כנגדו , ושהנשיא 
ואנשיו באמת מבקשים לרדוף אותו . לפני הנאום , אומרים פקידי ממשל , נעשו 
ניסיונות מאומצים לשכנע את המלך כי אין כל כוונה להתנקם בו או לפגוע בו בשל 
העמדה הנייטרלית , במרכאות או בלי מרכאות , שהוא נוקט . " הוא נמצא במצב 
קשה מאוד שם " , אמר בוש יומיים אחרי פרוץ המלחמה , על ידידו הוותיק . 
הוא הבטיח ש"לא למחוק את ירדן " כמי שיכולה " להמשיך להיות מדינה בעלת 
חשיבות עצומה בסדר הבין - לאומי החדש " . הנשיא הוסיף שלמרות המחלוקת שלו 
עם המלך שוררים ביניהם יחסי כבוד , והזכיר שלארה"ב היו מאז ומתמיד " יחסים 
ותיקים עם חוסיין " . הקש ששבר את אמונו של המלך בוואשינגטון , להערכת פקידים 
בכירים במשרד החוץ האמריקאי , היה פרשת רכישת הנפט של ירדן מעיראק . רבת 
עמון ביקשה , מאז תחילת המשבר במפרץ , שארה"ב תלחץ על סעודיה לספק לה נפט 
כדי שירדן לא תיאלץ להמשיך ולקבל את האספקה השוטפת מעיראק . אספקת הנפט 
מעיראק נתפסה בעיני רבים כהפרה של האמברגו המסחרי על עיראק . המלך הבהיר 
לוואשינגטון כי אין לו ברירה אלא לייבא את הנפט מעיראק , והציע שוואשינגטון 
תנסה לשכנע את הממשל הסעודי לדאוג לאספקת הנפט לממלכתו . אלא שהסעודים , 
בכעסם על עמדותיו הפומביות הפרו - עיראקיות של המלך , התנגדו . הממשל האמריקאי , 
באמת ובתמים , ניסה לשכנע את הסעודים לשתף פעולה , אך נתקל בסירוב . 
ממשל בוש החליט שלא להכביד על המלך , והכריז כמה פעמים כי בעיניו ירדן אינה 
מפירה את האמברגו אם היא מצטיידת , בלית ברירה , בנפט עיראקי . אחרי שפרצה 
המלחמה החלו ההפצצות של מטוסי הקואליציה על שיירות האספקה בכביש המוביל מעיראק 
לירדן , דרך האזור הנקרא בפי כתבי המלחמה האמריקאיים " ארץ הסקאדים " , 
בשל המשגרים המוצבים באזור ומופנים נגד ישראל . לטענת דוברים רשמיים אמריקאיים , 
היו מכליות נפט ירדניות משולבות בשיירות של משאיות אספקה צבאיות של עיראק , 
ולכן הן נפגעו מההפצצות . האמריאים הכחישו שהם ניסו לפגוע במתכוון במכליות 
הירדניות , אך המלך , כך סבורים בממשל , היה משוכנע שההפצצות נעשו בכוונה 
משום שהן נתפסו כמי שמפירות את העיצומים הכלכליים על עיראק . הפרשה הזו 
עמדה ברקע נאומו חוצב הלהבות של המלך , ובממשל הוחלט שאי - אפשר עוד להבליג . 
בייקר היה הראשון , בעדותו בסנאט ביום חמישי שעבר , ששלף את המקל ואיים 
בהקפאת הסיוע הכלכלי לירדן . עם זאת , הוא אמר , שיש להתחשב בלחץ שמופעל 
על המלך , ובכך ש"כאשר אנו מתבוננים סביב ובוחנים את החלופות למלך , התמונה 
אינה נאה כלל ועיקר " . ממשל בוש שרוי אפוא בדילמה קשה . הגישה השלטת במשרד 
החוץ האמריקאי עדיין אומרת שיש לתת למלך גיבוי , תהיה עמדתו הפוליטית בתקופת 
המלחמה אשר תהיה , משום שאחרי המלחמה הוא יהיה מרכיב חשוב בסדר המזרח - 
תיכוני החדש , שיתבסס על פרגמטיזם מדיני . הגישה המנוגדת אומרת , שהמלך 
הולך ומאבד את ערכו כבלם בפני הקיצוניות הלאומנית הערבית והקנאות הפונדמנטליסטית , 
משום שבהדרגה הוא הופך כלי משחק בידיהן . אם הרחוב ממילא מולך במלך , 
אולי אין טעם עוד לעזור לו . קריקטורה בואשינגטון פוסט הראתה ביום ששי 
שעבר את חוסיין כבובה בידו של סדאם , שדיבורה בוקע למעשה ממעמקי בטנו של 
הפיתום העיראקי . אם חוסיין אכן ייתפס כך בוואשינגטון , הוא עשוי לחוש את 
מכת המקל ולמצוא שהסיוע הכלכלי לארצו ( 53 מיליון דולר סיוע אזרחי ו02 מיליון 
סיוע צבאי ל1991 ) קוצץ עד כאב . חמור מזאת , הוא עלול להפסיד סיוע כספי 
נוסף בעתיד . ירדן פירסמה באחרונה מחקר כלכלי הצופה השנה הפסדים בשיעור 
של 83 . 1 מיליארד דולר כתוצאה מהמשבר במפרץ . המחקר פורסם במטרה לעגן בקשות 
לסיוע כספי נוסף . עם נטל של חוב חיצוני בסך 4 . 8 מיליארד דולר המוטל כריחיים 
על צווארו , עלול הממשל ההאשמי לקרוס בלחץ של מהומות ותסכול עממי . בממשל 
בוש שוררת תחושה שמחד גיסא יש לאיים על המלך בדרך כלשהי כדי להניא אותו מחציית 
הקווים אל המחנה הקיצוני בעולם הערבי , ומאידך גיסא קיים החשש שמימוש האיום 
הזה עלול להחיש בסופו של תהליך את נפילת הממלכה .
