התנועה המתקדמת , ההתארגנות הלאומית - הערבית הראשונה שהצליחה להשיג ייצוג 
בכנסת משנת 4891 ( בשותפות עם קבוצת אלטרנטיבה של מתי פלד ) נתונה במשבר 
עמוק . הוויכוח הפנימי מאיים על שלמותה ורבים מפעילי התנועה חוששים מפני 
פילוג . למעשה , הועלתה הצעה כזאת בכנס פעילי סניף נצרת , ובעלי ההצעה 
חזרו בהם רק כאשר מועצת הסניף הסכימה עמם , פה אחד , לקרוא להקמת ועד ארצי 
חדש , שייבחר בעיקר בסניפים , להכנת ועידת המפלגה האמורה להתכנס ביולי 
הקרוב . להלכה , אפשר היה לצפות לעליית כוחה של המתקדמת בימים אלה של התפוררות 
הגוש הקומוניסטי והשפעתה על חד"ש - רק"ח . אבל ראשי המתקדמת לא הצליחו 
לנצל את ההזדמנות . להיפך . דווקא בתקופת חולשה זאת של רק"ח שותקה הפעילות 
של המתקדמת . שבועון התנועה אינו מופיע זה חודשים . משרדי התנועה ומועדונה 
במרכז נצרת עברו לבניין משרדים באזור המוסכים , ואין בהם טלפון . ישיבות 
מוסדות התנועה נעשו זירות לסכסוכים , והעיתונות מלאה ידיעות על הפילוגים 
בתוך התנועה . החלטת נציג התנועה בכנסת , מוחמד מיעארי , לכונן רשימה משותפת 
לבחירות להסתדרות עם חד"ש ודראושה , היא המקור לסכסוכים . רבים מפעילי 
המתקדמת חשו נבגדים כשמצאו את עצמם בעלי ברית עם הקומוניסטים , שלא חסכו 
בביטויי גנאי ועלבון לאנשי המתקדמת במשך שנים . נראה היה , כי שנאת הקומוניסטים 
היא הדבק המלכד פעילים אלה במסגרת המתקדמת . הסכסוך בין רק"ח והמתקדמת גרר 
סכסוכים מקומיים בבחירות המוניציפליות ( כמו בכפר טירה ) ומשפטיים ( בין 
עו"ד כאמל דאהר , דובר המתקדמת לשעבר , ובין ראש עיריית נצרת ) . שני 
הסניפים העיקריים של המתקדמת במשולש טירה וערה שותקו , ופעילים מרכזיים 
בהם פרשו מפעילות . גם פעילים מסניף נצרת , ובראשם עו"ד דאהר , ממייסדי 
התנועה הבולטים , הקפיאו את קשריהם עם המתקדמת ואת פעילותם בה . הקומוניסטים , 
החרדים מעליית התנועה המוסלמית ומהיחלשות התנועות הקומוניסטיות בעולם , 
רצו ברשימה המשותפת כדי לעבור את אחוז החסימה , ובעיקר כדי לחסום את ההתארגנות 
האנטי - קומוניסטית של המתקדמת , דראושה והתנועה המוסלמית . גם הנהגת המתקדמת 
ודראושה רצו ברשימה המשותפת , כדי לזכות בייצוג במוסדות ההסתדרות ובתקציבי 
מימון המפלגות , וכדי למנוע מאלמנטים המקורבים לאש"ף בישראל ובשטחים לטעון 
כי הם מפלגים את הציבור הערבי , כפי שהיה במערכת הבחירות לכנסת ה21 . קפמאבינאבל 
גם הקשרים עם המפלגה הדמוקרטית - הערבית של ח"ך דראושה לא הוסיפו נחת לאנשי 
המתקדמת . נציגיהם במו"ם לכינון הרשימה להסתדרות הסכימו לקבל מעמד שווה 
לרשימת דראושה : % 02 לכל רשימה ו%06 לחד"ש . אנשים במפלגה טוענים , כי 
על - פי תוצאות הבחירות לכנסת היו נציגי התנועה צריכים לדרוש יותר והחלוקה 
היתה צריכה להיות פחות מ%85 לחד"ש , יותר מ%32 למתקדמת , ופחות מ%91 לדראושה . 
בבחירות לרשויות המקומיות התגלתה חולשתו של דראושה , כאשר רשימתו בנצרת 
לא עברה את אחוז החסימה ועיקר קולותיה נדדו לתנועה המוסלמית . גם החולשה 
של חד"ש היתה גלויה , נוכח התמוטטות המשטרים הקומוניסטיים במזרח - אירופה 
והתערערות מעמדה של רק"ח . נראה כי ההסכם לקראת הבחירות בהסתדרות היה מכת - 
מוות אידיאולוגית לרבים . מצד אחד היו צריכים לחתום על הפסקת אש עם הקומוניסטים , 
ומצד שני היה עליהם לקבל מעמד שווה למפלגה של איש אחד , שכמעט אינה קיימת 
בשטח . אפילו באיכסאל , כפר מולדתו של דראושה , זכה המערך לרוב קולות חברי 
ההסתדרות בבחירות האחרונות . בבחירות לכנסת זכתה חד"ש ב%14 של הקולות , 
דראושה זכה בכ%04 נוספים , והמתקדמת בשני אחוזים בלבד . יחד זכו ב%38 . 
בבחירות להסתדרות היה חלקם של כל מרכיבי הרשימה " המשותפת " ( חד"ש , 
המתקדמת ודראושה ) פחות מ%04 . המערך קיבל למעלה מ%14 . על רקע זה התכנסו 
פעילי המתקדמת לאחר מערכת הבחירות והחליטו , כי יש להכין ועידה חדשה שתזרים 
דם חדש למוסדות התנועה . הפעילים החדשים במוסדות קיוו כי ייערך בדק - בית 
בהנהגה , אולם במהלך ההכנות לוועידה התלהטו הרוחות . הטענה הרווחת כיום 
בתנועה היא , כי בוועידת התנועה הראשונה , בקיץ 7891 , השתלט ח"ך מיעארי 
על התנועה והרחיק ממנה מתחרים פוטנציאליים . אבל מיעארי לא חיכה שידיחו 
אותו . הוא וחבריו החליטו על בדק - בית בלשכה המדינית , שהביא להרחקת שניים 
מבכירי התנועה באזור המשולש אחמד מנצור מטירה ובדר יונס מערערה . גם שני 
חברי הלשכה המדינית מנצרת ריאח אבו אל - עסל וגמאל קעואר מצאו את עצמם 
מחוץ ללשכה ( אבו אל - עסל , מזכ"ל התנועה , חזר ללשכה עקב התפטרות אחד 
הנבחרים החדשים ) . אלא שההדחות החדשות , והעובדה כי הורחקו אחרוני החברים 
הנוצרים בלשכה המדינית , נתנו תחמושת בידי האופוזיציה שהאשימה את הנהגת 
התנועה בנטיות עדתיות . סניף נצרת , הכולל את מייסדי התנועה , סער . 
אחרי כמה פגישות החליטו פעילי התנועה פה אחד להצביע אי - אמון בהנהגה ולדרוש 
ממנה להפקיד את ההכנות וניהול הוועידה החדשה בידי גוף חדש , ש9 מבין 11 
חבריו ייבחרו על - ידי סניפי התנועה והם יוסמכו לקבוע את הצירים לוועידה . 
האם תמחל ההנהגה על כבודה , תקבל את האתגר ותתמודד על הרוב בוועדה המכינה 
של הוועידה , או שתדחה את הצעת סניף נצרת ? הדברים עומדים על סדר - היום , 
ויוכרעו בתוך ימים אחדים .
