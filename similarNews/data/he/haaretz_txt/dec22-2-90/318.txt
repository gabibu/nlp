הרכבת מרושטוק היתה אידיאלית : היא מגיעה לתחנת הרכבת המרכזית של לייפציג 
ב55 : 0 . הנווד העירוני רולי בן ה06 ושאר חסרי הבית ברציף יכולים אפוא 
להתענג על לילה שקט וחם . רק ב54 : 51 יוצאת הרכבת שוב לדרכה , ועד אז 
היא משמשת כמקום לינה . אמנם נוהגים לכבות את האורות ברכבות אלו המועברות 
למסילות צדדיות , אבל ההסקה פועלת והדלתות אינן ננעלות . בדרך כלל מעלימה 
משטרת התחנה עין . מאז ירדו מעלות החום בלילה אל מתחת לאפס , שינה תחת 
כיפת השמים עלולה להיות קטלנית לחסרי הבית , שלא אחת הם חולים ולעתים קרובות 
הם שתויים . ראלף בן ה92 התלונן : " שמה במערב אז תה יכול ללכת לכנסייה , 
אבל כאן כלום . לייפציג זה שיקגו . מכות , פריצות וגניבות , הן עבורו 
מעשה של יום ביומו , כמו גם ארוחה משיירי הטוגנים הקרים והשמנוניים בדוכני 
החטיפים בתחנה : " גם אנחנו רוצים לחיות . אם לא יקרה מהר מאוד משהו , 
השוטרים לא יספיקו יותר להסתדר עם המצב " . קפמאבינרשמית לא היו קיימים 
חסרי בית במזרח - גרמניה . משום כך אין במדינה שירותי רווחה מאורגנים כמו 
מעונות או מקלטים . הבעיה מצאה את היישובים במדינות הפדרליות החדשות בלתי 
מוכנים לחלוטין . השלטונות עומדים באוזלת יד מול התופעה , מה עוד שהקופות 
ריקות . מדי יום הולך ועולה מספר האנשים במזרח - גרמניה לשעבר , שתחילה 
מפוטרים מהעבודה , וזמן קצר לאחר כך מושלכים לרחוב על ידי בעלי הבתים שאצלם 
שכרו דירות . עקומת המצוקה תלולה יותר מאשר בחלקה המערבי של ברלין , שגם 
בה יש כיסי מצוקה רחבים . בברלין חיו בסוף 8891 כ005 , 5 איש ברחוב . בינתיים 
הגיע מספרם ל006 , 6 , ויותר מ002 , 1 מהם הם קטינים . " אצלנו עוד דברים 
רבים מתחוללים בחשאי , מודחקים " , אומר היינריך סונסאלה , מנהל לשכת הסעד 
במגדבורג . חסרי בית רבים אפילו לא ידעו לאן עליהם לפנות . סונסאלה חושש 
שגל של מצוקה יכה את הערים במזרח כאשר שכר הדירה יעלה לשיעורים מערביים , 
ובו בזמן יזנק אל על מספר הפועלים המפוטרים ממפעלים בלתי רווחיים . מאז 
האיחוד , הצטרפה אוכלוסייה חדשה לרחוב , שבמערב כבר מנתה שיעור גבוה של 
מחוסרי בית . בעבר דאגה המדינה המזרח - גרמנית , במסגרת התפקיד שנטלה על 
עצמה " לחנך מחדש " את עציריה , לספק להם מקום עבודה ודיור כלשהם לאחר 
השחרור . כיום איש אינו עוזר לאסירים המשוחררים . " הם באים מסוציאליזם 
מסודר , עתיר כללים ותקנות " , אומרת אווה שינאר , עמיתה של סונסאלה , 
" ואמורים להחזיק מעמד ולמצוא את דרכם בתנאי קפיטליזם מוקדם פראי ביותר " . 
אסירים לשעבר רבים כמהים כבר אחרי כמה ימי חירות לחזור לתא הכלא . אסיר 
ששוחרר במגדבורג , סיפרה שינאר , חטף שש פעמים תיקי יד ופריטים דומים אחרים 
כדי שיוכל להיתפס ולחזור לכלא , אך ללא הועיל . לבסוף , בייאושו , הוא 
העלה בניין באש , ומיהר להזעיק את מכבי האש לבל ייפגע איש . כעת , למשך 
שנה וחצי , תהיה לו קורת גג מעל ראשו וסורגים בחלונו . ברחוב גם מגיעה 
לקצה הקריירה של אזרחים רבים לשעבר ממזרח - גרמניה שברחו למערב מלאי חלומות . 
לאחר שנכשלו שם , הם נזכרים בביטחון הסוציאלי שהיה להם בבית על אף כל מגבלותיו 
ודלותו וחוזרים בלי לדעת מה מצפה להם . חזרתי הביתה אז עזרו לי , הם אומרים , 
והם משתאים על שאיש לא יכול עוד לעזור להם " , התלונן קלאוס הינצה , עד 
לא מכבר דיאקון לעניינים סוציאליים בכנסייה הפרוטסטנטית בלייפציג , וכעת 
חבר עירייה הממונה על בעיית חסרי בית . אפילו משפחות עם ילדים קטנים מגיעות 
לבתי מחסה ללא חשמל , ללא הסקה , ויכולות היו לחגוג את חג המולד רק עם 
נר על ארגז פירות . " מובן שקיימים חוקים סוציאליים על כל פרטיהם ודקדוקיהם " , 
אמר בזעם הינצה . " אבל אין לנו לא אמצעים ולא אנשים כדי ליישם אותם , 
כדי למתוח את רשת הביטחון החברתית " . קמאבינבקורסים מהירים מתכוונת אוניברסיטת 
לייפציג להכשיר כעת מורים לעבודה סוציאלית . אבל הינצה אומר , שכאשר הללו 
ייצאו לעבודה יהיה פיאסקו . " במקרה הטוב ביותר הם יוכלו לנהל את העוני 
והדלות " . הוא מעדיף את עוזריו לשעבר בקהילה סנט מיכאליס . להם יש לפחות 
ניסיון עם חסרי גג . מאז 7891 יכלו אסירים לשעבר ונידחים ופליטים חברתיים 
אחרים לקבל ייעוץ במרתף של הכנסייה , והמדינה לא התערבה . בינתיים מועמד 
בית התה , " די קוואלה " , פעם בשבוע לרשות כ002 חסרי הגג בלייפציג שרבים 
מהם פונים קודם כל אליו בניסיון להשיג עזרה . עוזריו של הינצה נותנים להם 
פרוסות לחם מרוחות בשומן וכוסות תה , ואז מסייעים להם במאבק עם הרשויות . 
שכן רבים מהם איבדו הכל , אפילו את תעודותיהם . וללא תעודת זהות הם אינם 
מקבלים את 004 המרקים , דמי הסעד לחודש שלהם הם זכאים , ואת התרופות שרבים 
מהם זקוקים להן בדחיפות . מעל הכל יש להקים במהירות האפשרית מקלטים לחסרי 
גג , סבור הינצה . אבל בשום מקום אין רואים יותר מאשר התחלות מהוססות מאוד . 
שכן הערים חייבות במימון מקלטים אלה וחסר להן כסף לכל מטרה ותכלית אפשריות . 
המעון הראשון במזרח - ברלין מורכב מקומץ דירות ששכרה לשכת הסעד . במגדבורג 
הוכנסו כ04 מיטות לאגף ההנהלה של בית הכלא לשעבר של השטאזי . בלייפציג מנסים 
עכשיו בעלי מלאכה להקים בית לפני שיבוא הכפור במלוא עוזו . במעון זה , 
הראשון מסוגו בעיר , יהיו 84 מקומות . זה מעט עד לגיחוך , יודע הינצה , 
אבל " המצוקה תאלץ אותנו למצוא פתרונות נוספים " .
