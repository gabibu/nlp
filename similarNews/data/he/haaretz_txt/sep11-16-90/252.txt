ועדת החשיפה שהוקמה במשרד האוצר מציעה להחיל את הליברליזציה בסחר ואת החשיפה 
ליבוא מתחרה כבר ב1 בינואר 19 . במסמך פנימי שהוכן באוצר , קובעים חברי 
הוועדה כי מאז גובשו הסכמי הסחר בין ישראל לשוק המשותף , חלו כמה תהליכים 
כלכליים משמעותים במסחר הבין - לאומי , המחייבים היערכות חדשה של ישראל . 
בין היתר הם מציינים : " עלה באופן משמעותי משקלו של גוש מדינות דרום 
מזרח - אסיה כולל יפן , ונוצרו פערי מחירים במוצרים רבים בין גושי המדינות 
" הזולות " ( דרום מזרח אסיה , דרום אמריקה ומזרח אירופה ) לבין הגוש 
האירופאי והאמריקאי שמוצריו יקרים יחסית " . לדעת מחברי המסמך " ההתפתחויות 
האחרונות במזרח אירופה ישנו בהדרגה את מפת הסחר הבין - לאומי ואת מערכת היתרונות 
היחסים בעולם , ויווצרו תנועות של הון מוצרים ועובדים בין גוש זה לבין גושים 
אחרים " . קפידבינעל פי תכנית הוועדה : יבוטל לחלוטין הנוהל של רשיונות ייבוא . 
ההגנה על התוצרת המקומית תתבצע באמצעות פעולות פיסקליות בלבד . בפריטים 
בהם היה אסור יבוא ממדינות שלישיות יקבע שיעור מכס ראשוני גבוה יחסית שילך 
ויפחת אוטומטית בתוך שנים מספר . תינתן עדיפות בטיפול בחומרי גלם ומוצרי 
ביניים . במוצרים אלה תיקבע תקופת התאמה קצרה שלא תעלה על שלוש שנים וייקבעו 
שיעורי מכס ראשוני נמוכים יחסית . קביעת שיעורי ההגנה הנדרשים עם תחילת הפעלת 
התוכנית תיעשה על בסיס השוואת מחירים , ותהיה התאמה בין המחיר הצפוי לבין 
המחירים הקיימים . שיעור המכס ייקבע נמוך במעט מהנדרש משיעור זה . במוצרי 
צריכה שבהם יש חשש מיבוא בהיצף , ייקבע מכס קצוב שימנע יבוא בתנאים שלא 
יאפשרו תחרות . סיוע לתעשיות הנפגעות מהחשיפה ינתן בדרך של סיוע בתחומי תשתית 
טכנולוגית על פי קריטריונים כלכליים וספציפיים לכל מפעל . קפידבינבמקביל 
לביצוע תוכנית החשיפה , שעליה תופקד ועדה משותפת של התעשיינים , נציגי 
האוצר ונציגי התמ"ס , מציעה ועדת החשיפה של האוצר לקבוע מדיניות חדשה בתחום 
התקינה שתמנע שימוש במכשיר התקינה להגנה מינהלית על תוצרת מקומית . הוועדה 
מציעה לחוקק חוק יעיל למניעת ייבוא היצף או מסובסד באופן משמעותי . היא 
קובעת כי עם הפעלת החשיפה יחוזק אגף המכס בכוח מקצועי בנמלים , שיטפל בעיקר 
בבעיות שיועלו עם ביטול ההגנות המינהליות ושיעורי המכס הגבוהים יחסית . 
כמו כן , יטפל המכס בבדיקת טובין , סבירות מחירים וסיווג נכון של הטובין . 
אנשי האוצר מציעים גמלשנות את מדיניות התקינה כך שלא תוכל לשמש ככלי למניעת 
הייבוא . בכוונתם לא לחייב מוצרי ייבוא לעמוד בבדיקות תקן כאשר מוצרים דומים 
מתוצרת מקומית לא נדרשים לעמוד בהן , ולצמצם את רשימת התקנים הרשמיים לנושאים 
של בריאות ובטיחות בלבד . כמו כן מציעה הוועדה לאמץ תקנים אירופאים ובין - 
לאומיים כך שמוצר שעמד בבדיקת תקן בארץ המוצא שלו יוכר כמוצר תקני גם בישראל . 
עוד מומלץ , להסמיך מכונים ומעבדות נוספות לבצע בדיקות תקן , ובטווח הארוך 
להותיר בידי מכון התקנים רק את סמכות לקביעת התקנים ואישור מעבדת הבדיקה 
בלבד . לצורך זה מוכן האוצר לתקצב את מכון התקנים ולמנוע את המצב הקיים לפיו 
המכון עצמו הוא גורם אינטרסנטי הרוצה לבצע בדיקות רבות ככל האפשר כמקור למימון 
פעילותו השוטפת .
