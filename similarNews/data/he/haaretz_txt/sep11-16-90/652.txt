תייר שיצא באחת השנים האחרונות לסייר במערת לאסקו שבחבל דורדון בצרפת והתבונן 
בציורי - המערות המפליאים של האמן הפליאוליטי , בתאו ובסוס ובאיל שברישומיהם 
יכול היה להתגאות גם פיקאסו , לא ראה בעצם את הציורים הללו , ואף לא סייר , 
לאמיתו של דבר , במערת לאסקו המהוללת . אינני חד חידות . פשוט , לפני 
כמה שנים הבחינו אנשי אגף העתיקות הממונים על האתר התיירותי הנפלא מטעם ממשלת 
צרפת , בנזקים העצומים שנגרמו למערת לאסקו המקורית בשל ביקוריהם של מיליוני 
התיירים מכל קצות תבל : נשימותיהם של אנשים רבים כל - כך הזיקו לציורים 
הצבעוניים יותר משהזיקו להם מאות אלפי השנים שעברו עליהם במסתור , קודם 
שגילה אותם ארכיאולוג הפרה - היסטוריה , האב בריי . איום של ממש היה תלוי , 
לדברי המומחים , על האתר כולו , ונשאלה השאלה איך לשמרו בלי שיוסיף להינזק 
בשל תיירות ההמונים המודרנית . נמצא כאמור פתרון רב - דמיון : נבנה אתר 
הזהה בכל למערת לאסקו המקורית , והוא מרוחק כמה קילומטרים ממקומה הטבעי 
של המערה . הרפרודוקציה , אומרים , היא מושלמת : גם עינו של מומחה לציור 
הפליאוליטי לא תבחין שזו לאסקו של דמה , שהתאו , הסוס וכיו"ב הם מעשי - 
ידי משחזר בן ימינו . וכך ניצלה מערת לאסקו האמיתית מכליון . הרי לכם תושייתם 
של המוסיאולוגים הצרפתים . מצד אחר , אין להכחיש שהתושייה הזאת מעוררת אי - 
נוחות לוגית מסוימת . התיירים עושים דרכם מקצוות - תבל , תמורת כספים שלפעמים 
הם לא - מעטים , כדי לראות את המקור המהולל , ותחת זאת הם זוכים לראות 
רפרודוקציה שבעיצומו - של - דבר יכולים היו לצפות בה גם במקומות - מוצאם , 
נאמר בגן העירוני של טוקיו או של תל - אביב . מבחינה לוגית , למצער , 
אם גם לא מבחינה ויזואלית ממש , עשוי התייר לחוש עצמו מרומה על - ידי המוסיאולוגים 
הצרפתיים ואף לכנות את תושייתם בשם הולכת - שולל . מצד נוסף והרי אין סוף 
לצדדיה של בעיה זאת מבין גם התייר התרבותי את הצורך בשימור , אף אם זה 
שימור מפני עצמו , נשימתו וכו . מזה זמן מה מעיינים גם האנגלים בצורך ובאפשרות 
להקים בבואת - נייר לאתר התיירות הנודע ביותר , הסמלי כמעט הוא מופיע על 
פוסטרים נפוצים של לשכת התיירות האנגלית של האיים הבריטיים : האבנים הקדומות 
של סטונהנג במחוז וילטשיר . מקום תפילה , או פולחן כלשהו , קלטי או אף 
קדום מזה , של סטונהנג , נעשה למחוז חפצם של מיליוני תיירים , ואף מקום 
מפגש של כתות שונות העובדות לירח וכיו"ב . מה בדיוק הנזק שיכולה היתה תיירות 
המונים זו להסב לאבני - הענק , היוצרות ביחד כעין פסל מודרני מופלא ( הנרי 
מור , בין השאר , שאב מאבני סטונהנג השראה לפיסול שלו ) קשה להבין , 
אך המוסיאולוגים האנגלים חרדים לגורלו של האתר וחושבים על הקמת סטונהנג דמה , 
בקרבת האתר הנוכחי , הדומה לו בכל , אך עשוי פלסטיק . איך יקבלו עובדי 
הירח ושאר העכו"ם את גזירת המוסיאולוגים , והרי להם דרושה מעל לכל האותנטיות 
הגמורה של המקדש הקדום בשל החשיבות שיש למיקומו המדויק מבחינת השפעת קרני - 
הירח במילואו וכו ? יתכן שלכל אחד מאתנו , וגם לאנגלים , יש צרות גדולות 
מזו , אך בכל זאת יכול יסוד התרמית שבמבצעים משמרים אלה לעורר התמרמרות 
גם אצל מי שאין להם זיקה ישירה כל כך למאגיה של המקום ולאותנטיות של האתר . 
נראה שהמגמה הזאת להקים אתרי - דמה , להציג רפרודוקציות מושלמות שפגיעתו 
של מטורף זה או אחר ( כשהמדובר בציורי - מופת ) לא תהיה פטאלית כל כך לגביהן , 
תוסיף להתרחב , ועל מקומם של רוב ( כל ? ) אתרי התיירות המבוקשים ביותר 
יתייצבו בקרוב בבואותיהן של היצירות המקוריות . עד איזו מידה יכולה המגמה 
הזאת להתרחב ? האם אפשר יהיה , למשל , להקים ונציה מחומרים אחרים שתהיה 
סמוכה , או לאו דווקא מאוד סמוכה , לעיר האיטלקית המקורית , הקורסת כבר 
שנים תחת כובד משא התיירים ארבעה מיליון בשבוע מסוים של הקיץ שעבר ! וראשיה 
אחוזי - הפאניקה בוחנים הצעות חדשות לבקרים של שימורה ? כל ההצעות עד כה 
( הטלת מס מיוחד על הכניסה לעיר , איסור שהייה מעבר ל42 שעות וכיו"ב ) 
נדחו , וספק אם תימצא דרך דמוקרטית לשימורה של העיר מפני פלישות התיירים 
אין הם צריכים בהכרח להיות ואנדלים , אף שגם זה קורה לרוב אלא אם תהיה 
זו בנייה של ונציה חלופית . הרעיון מפתה את הדמיון , מזמן אליו את הרעיון 
הבורחסי של התוויית מפה מדויקת כל כך שתהיה זהה בכל , גם בגודל , לאתר 
הממופה , המקורי . מכאן קל כבר להפליג אל עולם של המדע הבדיוני , לחלום 
על עולם חלופי שבו יהיו כל האתרים למה להסתפק באתרים ארכיאולוגיים או מוסיאליים ? 
ואולי כל החפצים בכלל ( וכל האנשים , לאחר החלפת - איברים טוטאלית ) 
חלופיים , לא ניתנים לפגיעה , ועם זאת דומים בכל למקורות " הטבעיים " 
( בשלב זה ישנה המושג " טבע " את משמעותו ) . אני נזכר בהפלגת - דמיון 
שכזאת של מחבר ספרי המדע - הבדיוני הפולני , סטניסלב לם , שכתב על השלב 
בעתיד שבו יהיו אנשים שכל אבריהם הוחלפו לטובים יותר , עד שנשאלה השאלה 
הפילוסופית ( האונתולוגית , למען הדיוק ) שהיא גם שם הסיפור : " האם 
אתה קיים , מיסטר גונס ? " והרי בקרוב תיהפך הבדיה הזאת לאפשרות יותר 
מתיאורטית . ההתחלות הללו של בניית עולם חלופי ( בינתיים עולם לתיירים , 
אך בקרוב וגו ) מביאות אותי אל הרעיון או ההצעה שזה כבר מנקרת בראשי בבחינת 
תשובה לכל התוכניות הללו של הארצות רוחשות אתרי - התרבות להערים על התיירים 
הפוקדים אותן בהמוניהם . ארצות אלו אינן מוכנות לוותר על הממון הבא מן התיירות , 
ועם זאת הן ערות יותר ויותר לנזקים האדירים שמסב התייר לאתרים , בפרט , 
ולאקולוגיה של ארצותיהם , בכלל . האיטלקים שוברים זה כבר את ראשיהם בבעיה 
הזאת , כנרמז לעיל , אך לפני שבועות אחדים החליטה אפילו ממשלת בריטניה 
על הקמתה של ועידה לשיפור מצב התיירות , ולאו דווקא במובן זה שהשירותים 
שיינתנו לתייר יהיו טובים יותר , אלא הבעיה היא איך להרתיע את התיירים מלבוא 
בהמוניהם לאנגליה , ועם זאת לא להרתיע אותם כליל , שהרי ענף התיירות מזרים 
הון עתק לאוצר האנגלי . והנה דומני שבידינו לפטור את הוועדות האנטי - תייריות 
הללו מן הדילמה הטורדנית שלהן : כנגד אתרי - הדמה ששירותי התיירות החדישים 
מתכוונים להעניק לנו , נצא גם אנו למסעות - דמה , או כפי שקראה לכך לאה 
גולדברג בספרון החמוד שלה : מסעות מדומים . אכן , לאה גולדברג היא מקור 
ההשראה האמיתי של הרעיון הזה , אף שהמשוררת ודאי לא שיערה , שעה שכתבה 
לפני שנים רבות ( שנות ה04 , כמדומני ) את ספרה " מכתבים מנסיעה מדומה " 
( הופיע לפני כמה שנים מחדש במהדורה מצולמת ב"ספרית פועלים " ) , שגם האתרים 
שהיא הפליגה אליהם על כנפי דמיונה ייהפכו בתוך עשורים אחדים ל"אתרים מדומים " . 
על היתרונות שבמסעות המדומים על פני המסעות בעליל ( ? ) אין צורך להכביר 
מלים . חיסכון בטרחה הו , אימי המסע המודרני ! הו , זוועת נמלי התעופה ! 
והחיסכון הכספי הוודאי הם רק שני פרטים בתוך מכלול ההטבות הזה שמביא המסע 
שאתה עורך לארץ לפי בחירתך ( אתר וכו ) בעודך שרוע בכורסתך המרופטת , 
מסתייע בספר ( כפי שהציעה לאה גולדברג ) או בכלים חדישים יותר שלא היו 
מוכרים לה . כל זה ברור היה כבר מזמן , ואכן לא מהיום קיים בשפה האנגלית 
הניב שהוראתו " בעל מסעות - כורסה " . אבל עד כה מצאתי איזה פגם באורח המחשבה 
ההגיוני והפרקטי הזה שבבסיס רעיון המסע המדומה . מי שיצר המסעות בלבו ואני 
מודה בחולשה זו הוא בעל כורחו רומנטיקן , ורומנטיקן נגוע , בעל כורחו , 
באותה אמונה פטישיסטית במאגיה של המקום , בצורך הלא - הגיוני במקור המקודש , 
באותנטיות הגמורה ( ואפילו גם הוא אינו מסוגל להבחין בעליל בין מקור לרפרודוקציה ) 
. ובשביל אדם כזה יש יסוד מרתיע בעצם הרעיון של המסע בכורסה . אבל החלפת 
האתרים המקוריים באתרי - הדמה , נוסח לאסקו ובקרוב אולי גם סטונהנג ( וכו ) 
, הורסת את המחסום האחרון הזה בפני אידיאל המסע בכורסה : לאחר ככלות הכל , 
אפילו רומנטיקן גמור שומר על שריד של שכל ישר , די הצורך לא לרצות להיות 
מרומה פעמיים : לצאת למסע מודרני טרחני לעבר אתר מודרני מדומה . אכן , 
מוטב כבר להיפך : מסע מדומה לאתר אמיתי .
