בולגריה היא המדינה האחרונה בגוש הקומוניסטי הנכנסת למהלך שינויים פוליטיים 
וכלכליים בקצב איטי , זהיר ושמרני . פסלו של לנין מוצב עדיין בסופיה בקרבת 
מלון " שרתון " . דגלים קומוניסטיים עדיין מתנופפים מעל מוסדות המפלגה . 
אך אווירת החופש מתפשטת ואופפת את ההמונים . אספות המוניות של הכנסייה 
והאופוזיציה מתקיימות , לקראת הבחירות הקרבות . בשבוע שעבר , בחג הפסחא , 
התכנסו בחצות הלילה במרכז סופיה , ליד כנסיית נייבסקי והמאוזוליאום של 
דימיטרוב כ051 אלף איש עם נרות דולקים בידם . הם שרו והתפללו בפומבי לישו 
ולדמוקרטיה , בפעם הראשונה מזה 54 שנות משטר קומוניסטי . בולגריה נפתחת 
לשינויים ובהם גם לתיירות מישראל . סיור מקיף של עשרה ימים בארץ יפהפיה אך 
בלתי נודעת זו חושף לתייר הישראלי מקומות מעניינים , מרתקים , יפים , 
מהנים וזולים מאוד . מישורים צבעוניים ענקיים של שדות גפנים ומטעים בפריחתם , 
ובהם עמק השושנים המפורסם . אתרי סקי ועיירות " שווייציות " של פסגות 
מושלגות בבורובץ ובפמפרובו . בתי מלון חדשים ציוריים מעץ . כנסיות פרבוסלוויות 
מיוחדות בצבעים עזים וקישוטים מדהימים ביופיים , במעבר השופקה , במנזר 
כילה , ובמנזר בצקובסקי . אזורים הרריים בתוליים עם שפע נחלים ויערות . 
הריוויירה הענקית לאורך חוף הזהב וחוף השמש , בים השחור , בסביבות עיר 
הנמל וארנה . כפרים עתיקים ציוריים ובהם מוזיאון הכפר הפתוח החדש באטר . 
בתי קאזינו להימורים בווארנה ובסופיה . מרכז בריאות מודרני ענק עם מעיינות 
מינרליים , בהשקעה גרמנית , נחנך לאחרונה בעיר סנדנסקי בדרום מערב בולגריה 
ליד הגבול היווני - יוגוסלווי . קבוצות מאורגנות אינן זקוקות לאשרות כניסה , 
בניגוד למצב בצכוסלובקיה , פולין ובריה"ם . טיסה של כשעתיים ורבע מביאה 
את התייר מישראל לסופיה בירת בולגריה . עיר נקייה בעלת מיליון תושבים ושפע 
של אתרים גנים , כיכרות , פסלים ובנייניים המזכירים את וינה ופאריס . 
בסמוך לעיר , רכס הר ויטושה עם פסגות מושלגות , בתי מלון , אתרי סקי 
ושבילי טיולים . המאוזוליאום על שם דימיטרוב ; כיכר לנין וסביבותיה ; 
בית המלון היפאני ויטושה ; כנסיית הבצלים הצבעונית ; מרכז התרבות החדש , 
בגן מרהיב ובו היכל התרבות המודרני והמשוכלל אן - ד - ק , אתר ואולם שלא 
רואים כמוהו ברבות ארצות אירופה . האתרים שחובה לסייר ולבקר בהם בסופיה כוללים 
את גן הפסלים , שבו פעמונים מכל ארצות העולם ; כנסיית נייבסקי היפה ביותר 
בבלקנים ; בית הכנסת הנמצא עכשיו בשיפוצים . כ06 אחוז מתוך אוכלוסיה של 
כתשעה מיליוני בולגרים מתגוררים בכפרים . רמת החיים נמוכה , הארץ חקלאית 
ומפגרת . העם הבולגרי ממושמע , סובלני ומקבל את גורלו בהשלמה . פרט לגלריה 
לאמנות בסופיה וכמה בתי מלון , שבהם מופיעים דברי הסבר גם באנגלית , כל 
שאר השלטים ודברי ההסבר באתרים , בדרכים ובמוסדות , כתובים אך ורק בולגרית , 
באותיות קיריליות , דבר המקשה מאוד על התייר . השירות אינו עומד באמות 
המידה המקובלות בתיירות העולמית , פרט לכמה בתי מלון . יש מחסור במצרכים 
ויש תורים , אבל לא לתיירים . עבור דולר אחד מקבלים באופן רשמי שמונה וחצי 
לבה . עכשיו מתפתח שם שוק שחור . לא מומלץ " ליהנות " ממנו עקב חוסר 
יציבות , שטרות ישנים שיצאו מהמחזור וחלפנים רמאים . מותר להיכנס ולצאת 
מבולגריה עם מטבע זר ללא מגבלות . בשדה התעופה לא דורשים הצהרת מטבע . כוח 
הקנייה של הדולר גדול מאוד בבולגריה . מונית לנסיעות בעיר תעלה בין חצי 
שקל לשקל וחצי . ארוחת חוויה ברמת עילית , בבית המלון היפאני היקר ביותר 
בסופיה , בקומה 91 , שממנה משקיפים על כל העיר , תעלה לזוג 0121 דולר . 
במסעדות " בודפשט " או " פורום " במרכז סופיה אפשר לאכול ארוחה חמה , 
פשוטה וטעימה , כולל בירה , בסכום של 45 דולר לזוג . במרכז סופיה , 
ליד מלון " שרתון " ומרכז המפלגה הקומוניסטית , יש כלבו ענק בן ארבע קומות , 
" צום " , הדומה מאוד בחיצוניותו לכלבו כינוצנטה במילאנו . בכל מה שקשור 
למוצרים אין שום דמיון . בכלבו אפשר להשיג מפות יפות ועבודות רקמה בזיל 
הזול . כל שאר המוצרים , וביניהם דברי הלבוש , מזכירים את תקופת הצנע 
בארץ . אפשר לקנות חליפות ב0121 דולר . לא רחוק מהמוזיאון ההיסטורי בסופיה 
יש חנות לבגדי גברים , שבה אפשר לקנות חולצות תוצרת יוון תמורת 45 שקלים . 
בערבים אפשר לבלות , במחיר חצי דולר , בקונצרט , באופרה , באופרטה ובמסעדות 
עם תוכניות פולקלור . בימים אלה מוצגים בסופיה הרקוויאם של ורדי , " גבירתי 
הנאווה " והאופרטות " צארדס " , " ברון הצוענים " , ו"הרוזנת מריצה " . 
בכל הזדמנות כדאי ליהנות מהמאכלים המקומיים , כמו גבינת הקצקבל , שמחירה 
דולר וחצי לקילוגרם , מעדני שושנים והיוגורט הבולגרי . הדעה הרווחת היא , 
כי מי שרוצה להגיע לזקנה מופלגת , חייב לאכול יוגורט בולגרי העשוי על - 
פי " נוסחה סודית " המתבססת על החידק הקרוי בצילס בלגריס " . במהלך הסיורים 
בכל אזוריה של בולגריה , נתקלים לעתים תכופות בגברים ונשים , שעל - פי 
מראה פניהם , ניתן להבחין שמוצאם מווייטנאם ומקובה . אלה הם פועלים שנשלחו 
על - ידי ממשלותיהם לעבודה בבולגריה , לתקופות של כמה שנים , במסגרת עסקות 
כלכליות בגוש הקומוניסטי והארצות מהעולם השלישי , בתמורה לסחורות ושירותים 
שקיבלו מהבולגרים . מדובר בכ51 אלף פועלים המשמשים כבני ערובה כלכליים . 
בולגריה משופעת בעשרות פסלים ואנדרטות לציון אנשים , אירועים ותאריכים מהיסטוריה 
בת 0031 שנה , הכוללות 861 שנות שלטון ביזנטי ו384 שנים של שלטון טורקי 
מוסלמי אכזרי בימי העותומאנים . האנדרטה המרשימה ביותר לדעתי , נמצאת בהר 
המשקיף על העיר שומאן . אנדרטה מרשימה ומיוחדת נוספת , נמצאת בעיר סטארה 
זגורה . אנדרטה מעניינת של הפטריוט והמשורר בוטב בעיירה קליפה ועוד . במהלך 
הסיור בבולגריה מצאתי להפתעתי מוזיאונים מיוחדים במינם : מוזיאון הצחוק 
והסאטירה בעיר גברובה ; מוזיאון האדמה בסופיה ; מוזיאון השושנים בעיר קזנלק , 
בירת עמק השושנים , המייצר % 09 של חומר הגלם לתעשיית הבשמים הצרפתית . 
המוזיאון האתנוגראפי בעיר העתיקה בפלובדיב , ומוזיאון תשמישי קדושה נוצריים 
במנזר רילה . על שתי כנסיות ומנזר אחד אני ממליץ שלא לוותר . במעבר ההרים 
שופקה נמצאת כנסייה פרבוסלווית עם כיפות בצלים מוזהבות . צבעיה העזים , 
הקישוטים המיוחדים והצריח עם צלב נוצרי " הרוכב " על סהר מוסלמי , הפכו 
אותה לאתר דתי ואדריכלי מיוחד במינו . כנסייה פרבוסלווית יפהפיה בצורתה , 
בצבעיה ובקישוטיה נמצאת גם במרכז סופיה . היא נראית כתפאורה הוליוודית 
מקורית ומיוחדת . המנזר רילה , שבו 0001 חדרים לצליינים נחשב ליפה במנזרים 
הבולגריים . הוא נמצא במעבה טבעי בין שבילי הרים מפותלים ומיוערים . הדרך 
אליו דומה למעברי ההרים השווייצרים או לרוט נפוליאון , המובילה מניצה ברוויירה 
הצרפתית לגרנובל . דרך מתפתלת בטבע פראי מובילה את המטייל למנזר מסוג שאיננו 
מכירים . צבעים וקישוטים בולטים ועזים בכל זווית . בכל קיר משולב עץ . 
חוויה בלתי נשכחת לבקר במקום , כאשר מסביב הרים מושלגים . שלוש הערים הגדולות 
המעניינות כל תייר בבולגריה מלבד סופיה הן : וארנה בים השחור , פלובדיב 
בחלק הדרומי וטארנובו בחלק הצפוני . בווארנה עיר הנמל יש אקווריום מעניין , 
דולפינריום צנוע , מוזיאון הימיה ובית אופרה נהדר , בצבע ארגמן מיוחד . 
בית הארחה של הצמרת הקומוניסטית יהפוך עכשיו לבית מלון ברמת דה לוקס . 
בעבר היה זה ארמון מלכי בולגריה . בית הארחה זה עשיר באוצרות ריהוט עתיק . 
בית המלון " גראנד " , בעל חמישה כוכבים , עומד ברמה אחת עם כל בית מלון 
אקסלוסיווי ומודרני במערב אירופה . הוא כולל קאזינו , בריכות מחוממות , 
באולינג , סאונה וכל יתר התפנוקים . טארנובו , הבירה הבולגרית העתיקה , 
בנויה על מדרון הר הסוגר על העיר שהוכרזה כשמורת טבע . בתים צבעוניים , 
באווירה מיוחדת ורגועה . בפלובדיב על נהר המריצה , חובה לשוטט בסמטאות 
הצרות של העיר העתיקה היפהפיה , שבהן אפשר לראות דוגמאות מקוריות וצבעוניות 
של בתים עתיקים , כולל ביתו של המשורר הצרפתי לה מרטן . במקום נחשף רק 
לפני כשנה אדטדרון רומי מעל לכביש מרכזי והיום הוא משמש מקום תצפית על העיר 
כולה . כדאי לבקר גם במוזיאון האתנוגראפי הקטן בעיר העתיקה , וכן במוזיאון 
הארכיאולוגי המציג את תרבות טראקיה ואוצר זהב של 7 ק"ג . והנקודה היהודית : 
המסורת ההיסטורית מספרת , כי המלך הבולגרי , בוריס הראשון , נטה לגייר 
את כל הבולגרים ליהודים , בזכות יועציו הקרובים שהיו בני דת משה . מבצע 
מסיונרי זה נמנע ברגע האחרון על - ידי שני כמרים נוצריים מסלוניקי ביוון , 
בשם קיריל , שמנעו מהמלך את הדבר . על שמם של אחים אלה נקרא היום הכתב 
הקירילי , שהם היו ממציאיו . הכתב נמצא בשימוש בברית המועצות , בבולגריה 
וביוון . העם היהודי " הפסיד " אמנם תשעה מיליוני בולגרים שנשארו נוצרים 
פרבוסלווים , אבל לזכותם תירשם לעד הצלתה של הקהילה היהודית בעת הכיבוש 
הנאצי .
