בסיום ביקורו האחרון של שר המדינה בייקר בסעודיה , דיווחו כמה ממלוויו לעיתונאים 
כי ראשי בית המלוכה הסעודי מוכנים למלא תפקיד פעיל ביישוב הסכסוך הישראלי - 
ערבי . עוד קודם לכן , בשיחות שניהל השגריר הסעודי בוואשינגטון , הנסיך 
בנדר בן סולטן ( בנו של שר ההגנה הסעודי ) הנחשב לשר החוץ הבלתי רשמי של 
המלך פהד , עם מנהיגים יהודים בארה"ב הצביע זה על נכונות ארצו להשתתף באופן 
פעיל במו"ם לשלום עם ישראל בתנאי שמו"ם זה גם ידון בפתרון הבעיה הפלשתינית . 
גם מרטין אינדיג , ראש המכון הוואשינגטוני למחקר מדיני ( הקשור במוסדות 
יהודיים ) , שהוזמן לאחרונה לריאד , טען לאחר שובו לארה"ב שאישים סעודים 
בכירים אמרו לו שארצם עשויה לתמוך במו"ם בין מדינות ערב לישראל ולהתעלם מהתביעה 
הערבית המסורתית שישראל תישא ותיתן עם הפלשתינאים בלבד . לדבריהם , אם 
תיפתר הבעיה הפלשתינית תהיה סעודיה מוכנה להסתלק מתמיכתה בחרם הערבי . בעקבות 
הנאמר למעלה ודברים דומים שהגיעו ממקורות אחרים , גברה האמונה בישראל לגבי 
האפשרות שסעודיה , בתיאום עם מצרים , תהיה מוכנה להוביל את מדינות ערב , 
או לפחות את מדינות המפרץ הערביות , ואולי אף את סוריה , למו"ם לשלום 
עם ישראל . למרות האופטימיות האמריקאית הזהירה לגבי התפקיד שתמלא סעודיה 
בפריצת הדרך למו"ם עם ישראל , אין הדברים פשוטים כלל ועיקר . מדינאים ועיתונאים 
שנפגשו עם המלך פייצל המנוח ( נרצח ב5791 ) , ששלל באופן מוחלט את קיומה 
של מדינת ישראל , הזדעזעו מדבריו לגבי היהודים שנשמעו כ"אנטישמיות גסה " . 
בגישה זהה מחזיקות עד היום האליטות המודרניות , ועל אחת כמה וכמה חוגי 
השמרנים והעולמא הקנאים בסעודיה , הדוחים עקרונית את קיומה של ישות פוליטית 
יהודית בלב העולם הערבי - מוסלמי , והנאחזים בדברים הקשים המיוחסים לנביא 
מוחמד ביחס ליהודים . העולמא והשמרנים הסעודים , המתנגדים לכל השפעה מערבית , 
הסכימו לנוכחות מומחים אמריקאים ואירופים בארצם , בתנאי ש"הכופרים " לא 
יקיימו פולחן , לא ישאו סמלי דת נוצריים בממלכה ויקפידו על האיסורים החמורים 
של האיסלאם הווהאבי באשר לאורח חייהם . נוכחותם של כחצי מיליון חיילים אמריקאים , 
כולל נשים , בשולי סעודיה , היתה כה טראומטית עבור השמרנים הסעודים , 
שנדרש פסק הלכה ( פתווה ) שהצדיק נוכחות זו בגלל הצורך להגן על הממלכה 
הווהאבית . המלך פהד , שכיהן כעוצר מ5791 עד 2891 , שלט למעשה בסעודיה 
כבר מראשית תקופה זו בגלל מצב בריאותו של המלך חאלד . פהד נחשב לראש הפלג 
המודרניסטי והפרו - אמריקאי במשפחה השלטת . בניגוד לפייצל ולחאלד , לא 
היה פהד אדוק בדתו ונטה להסכים להסדר של שלום בסכסוך הישראלי - ערבי . אף 
שהנשיא סאדאת לא יידע אותו על כוונתו לבקר בירושלים באוקטובר 7791 , היה 
פהד מוכן להמשיך ולתמוך במדיניותו הפרשנית של הנשיא המצרי לגבי ישראל כל 
זמן שלא תחרוג מהקונסנסוס הערבי . כידוע , סאדאת המשיך במו"ם שהוביל ב9791 
להסכמי קמף דיוויד . בתקופה זו חל סחף חמור באמינותה של ארה"ב באזור כתוצאה 
מאי נכונותה של ואשינגטון לסייע לבן - בריתה , השאה של איראן , וכתוצאה 
ממדיניות זו . הסעודים , שהסיקו מהתפתחויות אלו מסקנות לגבי עצמם , הצטרפו , 
לא ברצון , לחזית המיליטנטית שלאחר חתימת הסכמי קמף דייוויד הביאה להחרמתה 
של מצרים בעולם הערבי . פהד , שעל מדיניותו הפרו - אמריקאית נמתחה ביקורת 
קשה בארצו , נאלץ " להעדר " מהממלכה הסעודית לכמה חודשים " מטעמי בריאות " . 
קפמאבינמעמדו של פהד ומעמד סעודיה בקהילה הערבית השתנו כליל כתוצאה מהפלישה 
העיראקית לאיראן בספטמבר 0891 . פהד ניצל את מלחמת איראן - עיראק לחיזוק 
מחודש של קשרי ארצו עם ארה"ב , וכבר בראשית המלחמה הסכים להצבת מטוסי אייווקס 
אמריקאיים בבסיס דהראן . צעד זה היה חריג , מאחר שמחשש לתגובת העולמא , 
האינטליגנציה הלאומנית הסעודית וארצות ערב , דחתה ריאד בעבר בקולניות כל 
אפשרות של נוכחות אמריקאית במפרץ הפרסי . בסתר ביקשה ריאד מוואשינגטון להבין 
את מצבה העדין ולקיים נוכחות " מעבר לאופק " כדי שתוכל לסייע לה במהירות 
בעת משבר . באוגוסט 1891 נקט פהד מהלך שחרג ממדיניות החוץ הזהירה וההססנית 
שהיא טיפוסית לממלכה הסעודית כשפירסם תוכנית לפתרון הסכסוך הישראלי - ערבי , 
שכללה הכרה עקיפה בישראל . אם כי ריאד ויתרה בפסגת פאז ( השנייה ) ב2891 
על תוכניתה המקורית והסכימה להחלטות שבהן הודגשה התביעה להכרה בזכויות הלאומיות 
של הפלשתינאים וייצוגם הבלעדי ע"י אש"ף , טען פהד בפגישות עם מדינאים אמריקאים 
שהוא ממשיך לתמוך בהסדר עם ישראל והכרה בה תמורת החזרת " השטחים הכבושים " 
ופתרון הבעיה הפלשתינית . במקביל פעל פהד בעקביות להחזרתה של מצרים ל"חיק 
האומה הערבית " , למרות הסכמי השלום שלה עם ישראל . ריאד היתה השושבינית 
של פסגת עמאן בנובמבר 7891 , ( בהשפעתה השתתפה בה סוריה ) , שהתירה למדינות 
ערב לחדש את יחסיהן המלאים עם מצרים . לאחר שהסתיימה מלחמת איראן - עיראק 
( 8891 ) , הסתבר שבגדד לא ויתרה על שאיפותיה לעמוד בראש הלאומיות הערבית . 
כן התכוונה עיראק לרשת את מקומה של סעודיה כמנהיגת אופ"ק ולהביא לעלייה 
חדשה במחירי הנפט כדי לשקם במהירות את כלכלתה ולחזק את צבאה . במארס 9891 
הוקמה " מועצת שיתוף הפעולה הערבי " שכללה את עיראק , מצרים , ירדן וצפון 
תימן , ושהיתה מכוונת , בין היתר , לשמש מנוף נגד חברות " מועצת שיתוף 
הפעולה המפרצי " שהנהיגה סעודיה , שבעקבות המשבר בשוק הנפט סירבו לסייע 
לשיקום כלכלת עיראק והקטינו במידה ניכרת את מענקיהן למדינות ערב העניות . 
פהד , שהיה אמור לבקר את הנשיא מובארק בסוף מארס , טס קודם לכן לבגדד ושם 
חתם עם הנשיא העיראקי על שני הסכמי אי התקפה ואי התערבות בעניינים פנימיים . 
ריאד , במידה רבה תחת לחץ עיראקי , נטשה בחודשים הבאים את המדיניות שלה 
ושל כוויית , שהיתה מכוונת למנוע את עליית מחיר הנפט ע"י ייצור מוגבר , 
ועברה למחנה אופ"ק שדרש להעלות את מחיר הנפט ולרסן את חברות הארגון שחרגו 
ממכסת הייצור שלהן . בוועידת ארגון יצרניות הנפט הערביות ( אופ"ק ) ביולי 
0991 , קיבלה עליה כוויית את הדין ובוועידת אופ"ק , מיד לאחר מכן , הועלה 
מחיר הנפט מ81 ל12 דולר החבית , וחברות הארגון התחייבו שלא לייצר נפט מעבר 
למכסתן . כידוע לא היה בכך די כדי לשנות את כוונות סדאם חוסיין לפלוש לכוויית 
ולבסס לאחר מכן את ההגמוניה של בגדד במפרץ , דבר שהוליך למלחמה , בראשות 
ארה"ב , בעיראק . לצד הידיעות ממקורות אמריקאיים רשמיים על נכונות המנהיגות 
הסעודית להגמיש את עמדתה בעניין המו"ם עם ישראל , נראה שמלחמת עיראק לא 
הביאה לשינוי בעמדת האינטליגנציה והשמרנים הסעודים לגבי ישראל . הדים לכך 
נשמעים בכלי התקשורת הסעודית , שבהם מכונה ישראל גם עתה " האויב הציוני " 
או " הישות הציונית " , וכן בשיחות של מבקרים זרים עם סעודים משכבות שונות . 
בשני מאמרים שפירסם בשבוע שעבר כתב הוואשינגטון פוסט , סטיב קול , בהסתמך 
על ראיונות שקיים עם סעודים בעלי רקע מגוון , הוא טוען שהמלחמה בעיראק לא 
תרמה לשינוי גישת בני שיחו כלפי היהודים בכלל וכלפי ישראל בפרט . נציגי 
האינטליגנציה הסעודית טענו באוזניו שוואשינגטון אינה צריכה לטפל בתהליך שלום 
ישראלי - ערבי , אלא ללחוץ על ישראל להחזיר את השטחים הכבושים ולפתור את 
הבעיה הפלשתינית . היו גם כאלה שהצהירו בפניו שיצירת הקואליציה האנטי - 
עיראקית שהביאה לשבירת כוחה של עיראק , היתה " פרי מזימה ציונית שבאה לשרת 
את האינטרסים של ישראל " . קפמאבינאת הדיכוטומיה בעמדה הסעודית לגבי ישראל 
ניתן להסביר על רקע אופיה האיסלאמי - ווהאבי של הממלכה והתייחסות אוכלוסייתה 
ליהדות בכלל ולישראל בפרט , שהיא פרי דעות קדומות וקיצוניות שטופחו במשך 
שנים רבות ע"י העולמא ומערכת החינוך הסעודית . המלך פהד ואחיו ( בני אותה 
אם ) , החברים בקבינט ובמועצה המועצה המלכותית ( סולטן שר ההגנה , נאיף 
שר הפנים , וסלמן מושל הבירה ופרובינציית ריאד ) וכן גורמים אחרים באוליגרכיה 
הסעודית , נכונים כיום , יותר מבעבר , לתמוך במאמצי ואשינגטון להביא למו"ם 
בין ישראל למדינות ערב , או עם חלק מהן . עמדה זו מושפעת מהתחזקות עמדת 
ארה"ב באזור ומניצחון המדיניות הפרו - אמריקאית של פהד . אך שליטי סעודיה , 
שיציבות משטרם מבוססת מאז ומתמיד על עקרון " התייעצות וקונסנסוס " , חייבים 
להיזהר שלא להכעיס את העולמא , השמרנים והאינטליגנציה בארצם , שעמדתם לגבי 
ישראל לא השתנתה . נראה , כי סעודיה לא תוכל להיות בגלוי חלוצת המאמצים 
להידברות ערבית - ישראלית למרות תמיכתו של המלך פהד במאמצי ארה"ב להביא לסיום 
הסכסוך הישראלי - ערבי . עם זאת יהיה פהד מוכן לסייע לכל מדינה ערבית שתוביל 
מהלך כזה , ובמיוחד סוריה , אם תהיה מוכנה לכך . בנוסף , קשה להאמין 
שסעודיה הווהאבית תתמוך בהסדר שישמר את אחדותה של ירושלים תחת שלטון ישראלי , 
ולו גם יעניק זה למוסלמים ולנוצרים פיקוח על המקומות הקדושים להם .
