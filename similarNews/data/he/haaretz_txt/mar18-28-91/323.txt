התבוסה שנחלה המפלגה השמרנית השלטת בבחירות - המשנה במחוז עמק ריבל השאירה 
אותה בהלם . למתחריה , הדמוקרטים הליברלים , המפלגה השלישית בבריטניה , 
שקיבלו % 3 . 84 מהקולות , לא היה חסר הרבה להשגת רוב מוחלט . מרוב מוחץ 
בבחירות הכלליות לפרלמנט ( % 9 . 06 ) ירדו השמרנים למקום השני ( % 4 . 
83 ) . כאשר נשאל אחד המנהיגים המקומיים מה גרם לשינוי בדעת הקהל , אמר 
שיש לכך " שלוש סיבות : מס גולגולת , מס גולגולת ומס גולגולת " . אכן , 
מסר יותר ברור לא יכלו הבוחרים לשלוח לממשלתו של גון מייגור . אם יתמהמה 
הרבה , יפסיד במלחמת הבחירות למרות ההילה והתהילה של ניצחון מלחמת המפרץ . 
עמק ריבל הוא מחוז חקלאי בצפון אנגליה , שרק שכונה של עיר שדה קטנה אחת 
נמצאת בתוכו . רוב תושביו הם בעלי חוות של בהמות . חלקם מתפרנסים גם מתיירות , 
שכן חלק מהאזור הוא שמורת טבע ורבים נוהרים אליה . אין שם עוני של ממש 
ולא בעיות חברתיות בלתי שגרתיות . במלים אחרות : מחוז שמרני מובהק , מן 
הסוג המהווה את עמוד השדרה של אחיזת המפלגה הזאת בשלטון זה 21 שנה . המחוז 
נחשב כל - כך בטוח , עד כי הממשלה השמרנית החליטה להוציא מתוכו את חבר בית 
הנבחרים , שהיה שר הפנים , ולהעלות אותו לבית הלורדים . מהלך של הענקת 
תואר לורד לחבר פרלמנט גורר אחריו בחירות חוזרות לבית הנבחרים , מהסוג המכונה 
בדרך כלל " בחירות משנה " . לא היה ספק לשמרנים שבמחוז הבטוח הזה שוב ייבחר 
איש שלהם . משאלי דעת הקהל נותנים לממשלתו של גון מייגור יתרון גדול על פני 
הלייבור . מאז נבחר מייגור , חזרו השמרנים והגיעו לרוב כמעט מוחלט בציבור 
הבריטי . על פי שיטת הבחירות הרובית הנהוגה , מובטח להם שיזכו פעם רביעית 
להרכיב ממשלה , מעשה ללא תקדים בחמישים השנה האחרונות . מייגור עצמו הוא 
ראש הממשלה הפופולרי ביותר מאז מלחמת העולם השנייה , כשווינסטון צרציל נהנה 
מתמיכה דומה . כ07 אחוזים מהאוכלוסייה סבורים שהוא ממלא את תפקידו על הצד 
הטוב . בכל זאת הצביעו הבוחרים בעד הכיס והנחיתו את המכה הזאת על השמרנים . 
הם הבהירו גם , שבעצם אינם מסויגים כל - כך ממפלגתם המסורתית , שכן אחרת 
היו מצביעים בעד הלייבור . ההצבעה בעד הדמוקרטים הליברלים היתה הצבעת מחאה 
ברורה . במשאלי דעת קהל ענו רובם שאילו היו אלה בחירות כלליות היו נותנים 
את קולם לשמרנים , אלא שנפשם קצה במס המאוס . קפמאבינמס הגולגולת הושת על 
תושבי בריטניה לפני שלוש שנים על ידי ממשלתה של מרגרט תאצר . זהו מס המוטל 
על הכל מס מגורים הנגבה במקום ארנונה מקומית על פי מספר הנפשות מעל גיל 
81 המתגוררות בדירה מסוימת . ההסברים של תאצר ושל שר האוצר שלה דאז , נייגל 
לוסן , היו שמדובר בתשלום בעד שירות . הרשות המקומית זקוקה לכסף כדי לספק 
שירותים עירוניים . אלה כוללים סלילת כבישים , ביוב , מים , פינוי אשפה 
וכדומה . כל השירותים האלה ניתנים לא על פי גודלם של נכסים , אלא על פי 
מספר הנפשות שבהם . אם גרים בדירה חמישה אנשים , הם מייצרים פי חמישה אשפה 
מאדם אחד . הם גם שותים פי חמישה מים . הרשויות הורשו לקבוע את שיעור המס , 
אך לא לשנות את העיקרון . רוב בעלי הבתים הביעו סיפוק , כי מעתה לא חל 
עליהם עוד מס על רכושם , והוא שולם על ידי הדיירים . גם הלייבור רווה נחת , 
מכיוון שהתנגד לחוסר השוויוניות שבמס החדש ותבע במפגיע לבטל אותו ולשוב 
לשיטה הישנה של ארנונה על פי שטח המגורים ושווי הנכס . מס הגולגולת היה 
בשבילו מתנת - חינם של השלטון , כי הוא נתן לו נשק פוליטי והעלה את סיכוייו 
לנצח . חוסר הצדק שבמס כזה מובן גם בלי ההסברים המעמיקים יותר של הלייבור 
ושל הדמוקרטים הליברלים , שגם הם מתנגדים לו בחריפות . אלמנה עשירה , 
הגרה לבדה בבית בן שבעה חדרים , משלמת על פי עיקרון זה ששית מפועל הודי 
עני הגר עם אישתו ועוד ארבעה בנים בוגרים או קרובי משפחה בחדר אחד באותה 
עיר ; סטודנט חסר הכנסה משלם אותו סכום שמשלם לורד באחוזה סמוכה . סבלו 
מכך במיוחד בעלי הכנסות בינוניות ומטה . זוג שלו דירה קטנה , או בעלים 
של חווה חקלאית עם שני ילדים , היו עשויים לשלם פי ארבעה או אף פי ששה 
ממה ששילמו קודם לכן על פי שיטת ארנונת המגורים . הסיבה לשינוי היתה כרוכה 
בעלייה המתמדת במסים המקומיים . הממשלה המרכזית של מרגרט תאצר החליטה להפחית 
את הקצבת כספים לשלטון המקומי . הרשויות נאלצו להעלות את הארנונות . במשך 
שמונה השנים הראשונות של שלטון תאצר עלתה ארנונת המגורים ברשויות המקומיות 
פי שניים וחצי יותר מן המדד . הדבר עורר ביקורת נוקבת כלפי השלטון , והפוליטיקאים 
החלו מחפשים מוצא . עד היום אין יודעים בבירור מי הוציא את מס הגולגולת מספרי 
ההיסטוריה . עד כמה שידוע , הופעל מס כזה באחרונה לפני 002 שנה בנסיכות 
נידחת כלשהי באירופה , בעוד שבספרים ללימוד כלכלה הוזכר המס בעיקר כקוריוז . 
איש לא האמין , שממשלה רצינית במאה ה02 תרצה לשוב ולהפעיל שיטה שבה נענש 
מי שמצליח בתחום הריבוי הטבעי . אך נראה שתאצר העדיפה כל דרך אחרת על פני 
הפניית כספים נוספים לרשויות המקומיות . לדעתה , כשיש עודף כסף בידי הממשלה 
אין להוסיף שירותים , אלא יש להפחית מסים כמס הכנסה ומס חברות . המעניין 
הוא , שהלייבור הצליח ליהנות ממס הגולגולת כפליים : מצד אחד הוא ממשיך 
להתנגד לו ומשמיע את קולו ברמה בגנותו , ומצד שני דווקא הרשויות של הלייבור 
הן הגובות את מס הגולגולת הגבוה ביותר . לדוגמה , עיריית וסטמינסטר השמרנית 
בלונדון גובה מס של 053 דולר לשנה לנפש , ואילו עיריית הארינגיי שבידי הלייבור , 
גם היא בלונדון , גובה מס של 051 , 1 דולר לשנה לנפש . הבדלים קיצוניים 
בלתי מוסברים ושאינם מסתברים מרמת השירותים העירוניים שניתנים על ידי שתי 
העיריות . קפמאבינהלייבור , שגובה את המס הגבוה ביותר , מצליח עם זאת להפנות 
את הביקורת כלפי השמרנים . בעיריות ובמועצות שבהן שולטים השמרנים הם הפכו 
מטרה לחצי ביקורת גם אם הם גובים מעט , בגלל אי הצדק החברתי המשווע הבולט 
במס לא - פרוגרסיווי זה . מאז מס הגולגולת הפסידו השמרנים בכל בחירות המשנה 
חמש במספר . שניים מן המקומות הפסידו ללייבור , כשהמקומות לא היו " בטוחים " , 
כמו עמק ריבל . שני מקומות הפסידו לדמוקרטים הליברלים , מתוך אחד מהם 
שממנו יצאו חברי פרלמנט שמרנים בכל אחת מן הבחירות של עשרים השנים האחרונות . 
העיירה איסטבורן שעל חופה הדרומי של אנגליה מיושבת באוכלוסייה קשישה , 
שחלק ניכר ממנה הוא גמלאים מהשירות הממשלתי . תושביה גרים בווילות קטנות , 
בעלות גינות , ולחלקם יש משקי עזר . הרוב השמרני שם היה עצום וכל - כך 
בטוח , שהמפלגה השמרנית העמידה גם בו אדם שהיה מועמד להיות שר . את מי 
שרוצים בו בממשלה מציבים כמועמד במקום שבו יש ביטחון שיזכה ; מועמד שאינו 
נבחר אינו יכול להיות חבר הקבינט . באנגליה , שלא כמו בישראל , אי אפשר 
לצרף לממשלה אנשים שאינם חברי בית הנבחרים . עם כל זאת נפלה גם איסטבורן 
לידי המפלגה הדמוקרטית כשנרצח חבר הפרלמנט השמרני על ידי המחתרת האירית . 
לגבי איסטבורן השתמשו השמרנים עדיין בתירוץ שהמועמד שלהם חלש ואינו משתדל 
מספיק , אך לגבי עמק ריבל אף לא העלו רעיון כזה . היה ברור לכל , שהמועמד 
בסדר , המפלגה תמכה בו והוא בכל זאת הושאר מאחור . גם סקרי דעת הקהל וגם 
התבטאויות התושבים עצמם לא הותירו הפעם מקום לספק שהסיבה היא מהותית . כשהפך 
מייקל הזלטיין את עגלתה של מרגרט תאצר לפני קצת יותר ממאה ימים , הוא חרת 
על דגלו את הסיסמה שיש לבטל את מס הגולגולת . השמרנים לא בחרו , כזכור , 
בהזלטיין , אלא בגון מייגור , המקובל גם על המחנה הימני של תאצר בתוך 
השמרנים . מייגור עשה מהלך מזהיר : הוא צירף את הזלטיין לממשלה והעמיד 
אותו בראש צוות שתפקידו לחולל רפורמה במס . קווי ההנחיה שלו היו מעורפלים 
בכוונה : אם היה מקבל הוראה לבטל , היה מרגיז את כל מחנה תאצר , והיא 
עצמה היתה מרגישה נבגדת . אם היה מקבל הוראה לא לשנות משהו מהותי לא היו 
באים על סיפוקם כל התובעים את סילוק המס כדי לא להפסיד את השלטון . בעקבות 
מפלת עמק ריבל הודיע מייגור , שעל הזלטיין וצוותו לסיים את העבודה בתוך 
שלושה שבועות , כדי שיהיה אפשר להביא משהו לציבור עוד לפני הבחירות לרשויות 
המקומיות , המתקיימות בחודש מאי . אין לצפות שהמס יבוטל כבר השנה ; עיבוד 
הפרטים עשוי להימשך חודשים רבים . אך הצעות והסבר לבוחרים על העקרונות שינחו 
את הממשלה לקראת השנה הבאה חייבים לבוא , שכן לכל המאוחר בקיץ 2991 יש לקיים 
בחירות כלליות לפרלמנט . אחרי מלחמת העולם השנייה הריעו ההמונים לצרציל , 
המנהיג המלחמתי הדגול , אך בבחירות הכלליות הראשונות אחרי המלחמה הצביעו 
בעד מתחרהו , קלמנט אטלי . גם אחרי מהפך זה היה לרוב השיקול הכלכלי זה 
שהכריע בבחירות הכלליות . במשטרים דמוקרטיים יודעים השליטים היטב , שיש 
גבול להתנהגות על פי אידיאולוגיה . לפעמים הציבור אינו מבין , ובגלל היבט 
מסוים באידיאולוגיה הוא מצביע בעד המפלגה המתחרה . ואם לא נמצאים בשלטון , 
ממילא אי אפשר לקיים את האידיאולוגיה . על פי סקרי דעת הקהל , גון מייגור 
הוא כיום ראש הממשלה הפופולרי ביותר שידעה אנגליה מאז ימי צרציל . אך פופולריות 
לחוד ועמק ריבל לחוד . הוא אינו רוצה להיות שותף לגורלו של צרציל בבחירות .
