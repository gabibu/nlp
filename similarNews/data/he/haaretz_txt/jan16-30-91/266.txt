ספקולציות : המאמרים והדברים שכתב וסיפר שוקן על אביו הם עדות לתפקיד הדומיננטי

באמצעות מרטין בובר , תועדה על - ידי בנו באהבה ובהערצה . אך אביו היה 
גם איש עקרונות נוקשה , מאמין מוחלט במשמעת , והבנים נפגשו עמו בזמנים 
קבועים . הוא השפיע על בניו הרבה ידע אך לא אהבה לפחות לא אהבה גלויית - 
לב . בבית אוסרני זה גדל שוקן כשם שגדלו ילדים רבים במאה הקודמת : ללא 
אותה מערכת יחסים רגשית שהיום אנו רואים אותה כצורך נפשי . כמו בני דורו 
לא קיבל שום חינוך מיני , אך מכיוון שגם לא שאב את השכלתו מהרחוב , מצא 
עצמו הולך לבית ספר כאשר הוא בור בעניינים אלה . בגיל שבו ילדינו מתגססים 
עם הוריהם , הוא האמין כי אישה נכנסת להריון עקב מבט חודר הנתקע בין עיניה . 
משום כך , במשך זמן ניכר , חשש הילד היוצא מהבית חמר - הסבר להביט במישרין 
בעיני אחת המורות שהעריץ . זהו פן אחד קטן של ילדותו , המאיר את העולם 
שעיצב את אישיותו . ילדות זו מלווה אותו עד גיל מאוחר , גם כאשר עקרה המשפחה 
לפרבר צלנדורף העשיר של ברלין , גם כאשר נסע ללימודים בלונדון , גם כאשר 
החל לערוך את " הארץ " שאביו קנה . אכן , סיפור חייו מתמצה במאמץ נמשך , 
המצליח חלקית , להשתחרר מאזיקי בית אבא הבית שממנו צמחה אימפריה של חנויות 
כל - בו , והבית של חוכמת ישראל וחוכמת אשכנז , הבית שבו ילדים גדלו בצל 
אבא ולא עם אבא . בתהליך השתחררות זה חלה תפנית כאשר נפטר זלמן שוקן ב9591 , 
והוא בן 28 . גרשום שוקן היה אז בן 74 ובעל משפחה ומעמד ציבורי , אך מות 
אביו האיץ בו תהליך של אמנציפציה אישית שבאה לידי ביטוי בעניינים שהם שגרתיים 
אצל אדם אחר . בגיל 05 למד נהיגה , ולראשונה בחייו אחז בהגה ונהנה כגימנזיסט 
צעיר מהחופש שהדבר היקנה לו . רבים מאתנו , צעירי המערכת באותם ימים , 
גילו , אחוזי פלצות מסוימת , כיצד המכונית החולפת על פנינו בשריקה נהוגה 
בידי לא אחר מאשר מר שוקן . בגיל מאוחר גם גילה את משה פלדנקרייז , למד 
ממנו את רזי השליטה בגוף , התפעם נוכח התוצאה המופלאה , והפך להיות חסידו . 
באותה תקופה עצמה מצא מנוחה וסיפוק בחיים משותפים עם רעיה שבמזגה היתה היפוכו 
המוחלט של בית אבא . תקופה זו כשלושים השנים האחרונות לחייו היתה גם התקופה 
שבה חיפשה סקרנותו הטבעית ומצאה לה אפיקים חדשים . לא היה נושא שלא היה 
מעניינו . הערצתו למשוררים הגדולים גרמנים ואנגלים כולם לא מנעה ממנו 
לנסות ולהבין מדוע זמר הרוק רוד סטיוארט עורר בביקורו בארץ התלהבות כה גדולה . 
הוא שמע על ספריו של לוי - שטראוס , על מחזותיו של יונסקו , ועל סרטיהם 
של פליני , ברגמן ופאסבינדר , והשקיע עצמו לתוכם בהתלהבות של איש צעיר 
ורעב . רישומיו של תומרקין מעטרים את ספר שיריו . אך יחסו לעולם האמנות 
המודרנית , שהשתנה ללא הרף לנגד עיניו המסוקרנות , לא היה חד - משמעי . 
שמרנות וחדשנות שימשו אצלו בערבוביה . הטענה המרכזית שהיתה לו נגד היצירה 
העכשווית היא שאין בה סגנון דומיננטי האופייני לתקופה ולמקום ; הסד שבו 
חייב היוצר לצקת את פרץ דמיונו אינו קיים . בשום תקופה קודמת לא היתה תופעה 
מסוג זה . עד למאה העשרים יכולת לזהות יצירה מוזיקלית , ויזואלית או ספרותית 
על פי סגנונה . האמנות העכשווית היתה בעיניו בליל שבו צו האופנה החולפת 
מחליף את הסד הסגנוני . וחוץ מזה , נהג להסביר , לא נתחדש באורח מהותי 
דבר מתקופת הניסויים הגדולה שקדמה למלחמה הראשונה , תקופה שבה הנועז ביותר 
הא - טונלי , האקספרסיונסטי , זרם התודעה הספרותי מבתי היוצר של גויס 
ופרוסט היה באמת אוונגרד . מה נתחדש מאז ארנולד שנברג וה"נערות מאוויניון " , 
שאל באורח רטורי . את הסגנון הפוסט - מודרני בארכיטקטורה ביטל באומרו כי 
" קיטש " היא מלה קצרה יותר ונכונה יותר . מה שהיה מודרני בשעתו הפשטות 
הפונקציונלית של הבאוהאוס , מנדלסון , גרופיוס , ברויאר נראה לו כקלאסיקה 
מודרנית שלא נמצא לה יורש ראוי . אך הוא היה מודע לחילופי הטעמים גם בעניין 
זה . כאשר הגיע לניו יורק לראשונה בשנות השלושים היה בניין קרייזלר סמל 
לכיעור מטופש תשלובת דוחה של סגנון " גותי " עם גורד שחקים ואילו היום 
הוא בניין קלאסי שיש לשמרו . אכן , עניין זה של שימור העיד על אותה שנית 
שהיתה בו לגבי ישן וחדש . הוא התנגד למה שראה כהיסטריה של שימור מבנים בארץ . 
מבחינתו ראוי היה לשמר רק מבנים שיש להם ערך אסתטי עצמאי , ואין שום מניעה 
להרוס ישן רק משום שאדם חשוב התגורר בו . עד כדי כך הגיע שלא רצה אפילו 
לשמר את ביתו של ש"י עגנון בתלפיות , ונשאר בעניין זה בדעת יחיד . לדידו , 
טוב היה בניין נאה בן ימיו ממבנה ישן ומכוער שיש לגביו סנטימנט היסטורי . 
את הגעגועים לתל אביב הישנה לשכונת מחלול , לקיוסק של רובננקו למשל 
הגדיר כתופעה חולנית וגרוטסקית . העבר היה חשוב תמיד , עוד מהימים שהיה 
עולה לרגל לביתו של גתה בוויימר , אך בצד הקלאסיקונים היו גיבוריו בני זמננו . 
ברשימת המשוררים האהובים עליו , שהציג בהקדמה לספר שיריו שיצא ב9691 , 
נכללים ט"ס אליוט , רוברט גרייבס , אודן , קוואפיס ושני משוררים שבקורותיהם 
וביצירתם היה ממש מומחה : ייטס האנגלי - אירי וגוטפריד בן הגרמני . כאשר 
באה הטלוויזיה וחדרה לעולמו מצא בה עניין רב , וגם אהב ורצה להופיע לפני 
המצלמה . ערבוב זה של שמרנות וחדשנות בא לידי ביטוי בכל . הוא השתאה והתפעל 
מהחופש המיני המאפיין את הדור החדש , אך תיעב את העיסוק הזול , הפומבי 
והוולגרי במין בעיתונות ובספרות . הוא טיפח כישרונות חדשים והכניס סגנון 
מנתץ איקונין לעיתונו , אך התקומם כולו נגד הסגנון הרכילאי , ה"כסחיסטי " 
המלה עצמה היתה שנואה עליו של מה שקרוי " ניו - גורנליזם " . עיתונות 
לא היתה בעיניו עוד עסק , אלא גם שליחות מחייבת . למעשה , קשה היה להגדיר 
תחום מתחומי עיסוקיו באורח חד - ממדי . השקפת עולמו היתה קפיטליסטית והוא 
הנחילה לקוראי " הארץ " גם בימים שזו היתה מלה נתעבת אך אחת מגיבורותיו 
היתה רוזה לוקסמבורג . היתה לו תיאוריה כי אילולא נרצחה , היו שונים פני 
הדברים בגרמניה . הוא היה סמל הרציונליזם החילוני והספקני , ובערוב חייו 
פיתח התנגדות עזה גם לכפייה הדתית וגם לנושאי דגלה של הדת בישראל . אך לאחר 
מבצע אנטבה כתב במאמר ראשי ב"הארץ " דברים אלה : " כבוד ותהילה לחיילי 
צה"ל ולצוות חיל האוויר שגבורתם עמדה לנו . . . והודיה לצור ישראל שהצליח 
בידיהם " . מורכבות זו העמידה אותו בניגוד ל"שטנציות " הישראלית , זו הרואה 
כל איש ציבור כבעל תווית היכר , המכתיבה מראש עמדות בנושאים שונים . ייתכן 
שתרומתו העיקרית היתה ב"לא - ישראליות " זו . סירובו העיקש להיות חלק מחיי 
החברה הישראלית להחניף לכולם , להתחבר עם כולם ואחר כך לרכל על כולם היה 
רק חלק מהעניין . הוא בלט באי - רצונו להיות חלק מהבליל הרגשי והשטחי המאפיין 
את חיינו הציבוריים , ולא פעם נהנה גם ל"הרגיז את הבורגנים " כמו בהוראותיו 
להפר החלטה של צנזורה , לא להיענות למשאלות שרים בעניין פרסום חומר חסוי , 
וכמו בהספד גלוי - הלב והביקורתי שכתב על גולדה מאיר ביום מותה . והוא היה 
גם " לא - ישראלי " מעוד בחינה : רק שמץ קטן מהאוצר הבלום , מבור הסד 
של ידיעותיו , הופגן כלפי חוץ . מי ששלט בעיתון , והיתה לו עמדה אישית 
בכל עניין תחת השמש , מיעט לכתוב . בחברה שבה רצים לספר לכולם על כל בדל 
של ידיעה שטחית , בעולם שבו כולם עושים גלים , והקצף מים לים מכסה , היה 
הוא הקרחון , שרק מעט ממנו מזדקר החוצה . גם " הארץ " שיקף תכונה לא - 
ישראלית זו , ונתן ביטוי למורכבותו האישית : שילוב של איפוק וחוסר ריסון , 
תרכובת מוזרה של הניו - יורק טיימס וה " . מורכבות זו לבשה אצלו גם דמויות 
שונות : היה מר שוקן זהשתואר מעל דפי " הארץ " לאחר מותו פורמליסט , 
קפדן , תובען , וגם , לא תמיד , נוקשה ביחסיו עם עובדיו . והיה גוסטב , 
אדם שונה לחלוטין , איש שיחה ומשקה סליבוביץ וויסקי שלא הסתיר את תאוותו 
לרכילות עסיסית ושאהב את מנעמי החיים . כאשר נשברו מחיצות יצר ברגעי חסד 
קשר אישי עמוק , לרוב קצר - ימים , שכן האכזבה באה באותה עוצמה שבאה ההתלהבות . 
והיה גם רוברט פוזן , האדם הנחבא במרתף עצמו , וממנו הוא משלח מדי פעם 
בפעם את שיריו בשלוש התרבויות שבהן היה בן - בית עברית , אנגלית , גרמנית 
( אני כותב בשלוש שפות , אמר , בגלל משה רבנו ואדולף היטלר ) . גם בשיריו 
יש שנית . הסגנון הוא קר ומנוכר וניכרת בהם השפעתו הגדולה של אודן אך 
התוכן נע בין חושניות יצרית למורבידיות רומנטית . באחד משיריו שנכתבו באנגלית , 
" ירושלים מאחור " , הוא מתאר את ירידתו ב"סרט האפל " לקראת אדמות השפלה 
כנראה , מהר הזיתים לעמק קדרון כאשר " מצבות חורי מעל ראשי עולות מעלה 
מעלה " . מצבת אמו , שאותה ממשש האב הקפדן כדי לוודא שאכן קיבל את מיטב 
הסחורה , מתוארת על - ידי גרשום שוקן באחד המאמרים שכתב על אביו ב"הארץ " . 
הפרסונות השונות לא היה נפרדות , אלא חופפות ומשיקות זו לזו . כאשר ישב 
מאחורי המכתבה היה הוא תמיד " מר שוקן " , בן - דמות אביו , הבוס , " השף " . 
אך כאשר רק קם ויצא למסדרון ראית מולך אדם שונה לחלוטין נמוך יותר ממה 
שחשבת , נע במהירות עצבנית , נמנע ממפגש עין ; לעתים היה נדמה לי כי צעדים 
אלה אינם אלא הד לצעדיו של הנער גוסטב , המוזמן לפגישה עם אביו , בבית 
שוקן אשר בצוויקאו . הוא נכנס לשמוע את אביו . " הוא אהב לקרוא שירים 
נזכר גוסטב כשהיה למר שוקן " הוא קרא באוזנינו שירה בגרמנית , וזו אחת 
החוויות החזקות של ילדותי . הוא קרא יפה בקול חמים , בריטוני . הוא חי 
את השירים האלה " . הנער גוסטב שמע שירה זו , באולם שמקירותיו ניבטים כרכים 
הדורים של שירה והגות , ובתוכו ניצבת הדמות הנערצה שבצלה חי , וממנה , 
בסופו של דבר , השתחרר . גרשום שוקן סיים את ההרצאה שנשא בספריית שוקן 
במלאת שלושים שנה למות אביו במובאה מתוך שיר של ט"ס אליוט למעשה דברים שהשמיעה 
מרי סטיוארט לפני מותה : בקצי ראשית דרכי . וסיום זה נכון גם לגבי גרשום 
שוקן . זכרונו לברכה .
