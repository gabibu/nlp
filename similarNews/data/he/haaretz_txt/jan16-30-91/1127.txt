בעידן הבנקאות האלקטרונית הופך המחשב לאמצעי ייצור חיוני , שבלעדיו אין 
לבנק תקומה . כבר בשבוע הראשון של המלחמה ישבו ראשי יחידת המחשוב של בנק 
מסוים מול ספקי ציוד ושירותים בתחומי הגיבוי וההתאוששות ממצבי אסון , בניסיון 
לגבש תוכנית מהירה להקמת מערך גיבוי אמין . גיבויים , אומר מאיר פורמנסקי , 
מנהל מרכז המחשבים של יבם ישראל , העוסק גם בייעוץ ללקוחות על התאוששות 
במצבי חירום , הם עניין שקשה לשכנע בכדאיות הכלכלית המיידית שבו . אתה 
משלם פרמיה גבוהה מבלי לקבל תמורה נראית לעין בטווח המיידי : קשה להוכיח 
שירות טוב יותר למשתמשים או זמני תגובה משופרים . מנהלי יחידות עיבוד נתונים , 
הכלואים בדרך כלל בסד תקציבי לוחץ , ייטו להשקיע בתחומים שתוצאותיהם נראות 
היטב לעין . " אחרי המלחמה הזאת אולי יתחיל מישהו לתת את הדעת לעניין הגיבויים 
במיתקני מחשב מרכזיים " , אומר אברהם חסון , מנכ"ל היטאצי דאטה סיסטמס ישראל . 
זה שנים אחדות מנסה חסון להציע שירותי גיבוי למיתקני מחשב מרכזיים . בין 
השאר , אף הציע עסקת ביטוח : תמורת 06 אלף דולר בשנה יעמיד לרשות הלקוח 
אפשרות שימוש במיתקן גיבוי נייד . אלא שאברהם חסון , ועמו רבים וטובים אחרים , 
לא מצא אוזן קשבת אצל לקוחות גדולים או במשרדי ממשלה . הרעיון להקים מרכז 
גיבוי ממשלתי מתגלגל אמנם תקופה ארוכה בין פקידי האוצר וראשי מערך המחשוב 
הממשלתי , אולם מלחמות תקציבים ולא פחות מכך חילוקי דעות בסוגיית האחריות 
למרכז הזה מנעו לפי שעה את הקמתו . הפגיעות הראשונות של טילי הסקאד העלו 
מחדש על סדר יומו של ענף המחשבים בישראל את בעיית הגיבוי וההיערכות להתאוששות 
במצבי חירום . עד עתה לא נגרם נזק משמעותי ולו גם למיתקן מחשב אחד . במפעל 
תעשייתי במרכז הארץ שבחצרו נפל טיל נרשמה תקלה במערכת ההפעלה במחשב , מתוצרת 
ן - סי - ר . הבעיה תוקנה באמצעות קו התקשורת שבין המפעל ובין מחלקת השירות 
בחברת י . א . מיטווך , נציגת ן - סי - ר בארץ . תקלות מלחמה במיתקני מחשב 
אינן חייבות להיות תוצאה של פגיעת טיל דווקא . הכיבוי התכוף של מערכות מיזוג 
האוויר , כמתחייב מהנחיות הג"א עם כל אזעקה , מחייב גם כיבוי מקביל של 
המחשבים עצמם , כדי שלא ייגרם להם נזק עם עליית הטמפרטורה בחדר המחשב . 
אלא שכל כיבוי כזה עלול ליצור תקלות דיסקים שקשה להתאושש מהן . יש להבחין 
בין שני סוגי פגיעה : נזק פיסי לחומרה , למחשב ולציוד ההיקפי עצמו או 
נזק לנתונים : תוכנה או רשומות . קל לתפוס את משמעותם הכלכלית של נזקי 
חומרה ; קשה יותר להעריך את משמעות מחיקתם של רשומות הנהלת חשבונות או ניהול 
הייצור . במקרים רבים עלול נזק למחשב לשבש לחלוטין את פעילותו של הארגון 
בין אם הוא נותן שירותים ובין אם מדובר במפעל יצרני . הנזקים העקיפים הנגרמים 
בשל שיבוש כזה אובדן הכנסות , פגיעה במוניטין ועריקת לקוחות לזרועות המתחרים 
מעמידים באור אחר את ההשקעות בגיבוי , יהיו גדולות ככל שיהיו . אינג אליהו 
מרון , יו"ר לשכת היועצים לניהול , אומר כי סקירה של מפעלים רבים שבהם 
קיימים מחשבים , ואשר קיבלו ייעוץ מחברי הלשכה על גיבוי ואבטחת מידע , 
מעלה כי על אף הנהלים הקיימים לא מקוימת השגרה של אבטחת מידע באופן מלא . 
כדי לצמצם את הנזקים העלולים להיגרם , מציע מרון לשמור העתק מעודכן של התוכנה 
באתר מרוחק ממקומו של המחשב רצוי בביתו הפרטי של אחד המנהלים או עובדי יחידת 
המחשב . בתנאים שבהם אנו נתונים , אומר מרון , חובה לעשות גיבוי יומי 
בסוף כל יום עבודה , ולשמור גם את הגיבוי הזה במקום מרוחק מהמחשב . העתק 
אחר של הגיבוי היומי יישמר במקום מרוחק , בתוך בניין המפעל , רצוי בכספת . 
אין צורך לומר , כי שני העותקים של הגיבויים היומיים חייבים להיות מעודכנים 
מדי יום . כדי להגביר את הביטחון , ממליצים המומחים לשמור כמה דורות של 
גיבויים . המהדרים מחזיקים בגיבויים יום - יומיים של חמשת ימי העבודה האחרונים , 
ובנוסף העתק גיבוי של יום העבודה האחרון בשבוע הקודם . באופן כזה , מסביר 
מרון , אפילו תתרחש תקלה גם בגיבויים עצמם , לא יהיה קושי רב בשחזור החומר 
העדכני . קפמאבינחומרי לחימה כימיים מעוררים בעיה מיוחדת גם כשמדובר במערכות 
מחשב . מדיה מגנטית דיסקים , קלטות וסרטים רגישה ביותר לחומרים אלה . 
המדיה המגנטית בנויה משכבת פלסטיק ומשכבה מגנטית . הדבק הקושר בין השתיים 
רגיש ביותר לחומרי לחימה כימיים . חברת י . א . מיטווך ובניו , נציגת 
יצרנית המדיה המגנטית 3 בישראל , פנתה אל ריימונד אנדרסון , מנהל המעבדה 
המרכזית הבין - לאומית למדיה מגנטית של 3 בסנט פול , מינסוטה , בבקשה לקבל 
הנחיות התנהגות . כל מדיה מגנטית , מתריע אנדרסון , ובעיקר זו המצויה 
בתוך כוננים , עלולה להיפגע אם תהיה חשופה לרטיבות ולרסיסי טיפות חומרי 
לחימה כימיים . בעת התקפת גז מציע אנדרסון לכסות בניילון את יחידות הדיסקים , 
סרטים מגנטיים וקלטות הגיבוי . בכל מקרה , אין להחזיר לכונן מדיה שלגביה 
קיים ולוא גם חשש קלוש שנחשפה לחומרים כימיים . המלחמה במפרץ הביאה לשיפורים 
מיידיים בנוהלי הגיבוי הנהוגים במלם , המרכז למיכון משרדי , המטפל בחלק 
נכבד ממערך המחשוב הממשלתי האזרחי . מערכות המרכז , הכוללות את מרשם התושבים 
וניהול סחר החוץ , מגובות במחשב שע"ם , המשרת את מינהל הכנסות המדינה . 
מאז פרוץ המלחמה מעבירים שני הגופים זה לזה מדי לילה את קלטות הגיבוי שלהם , 
בניגוד למצב הקודם , שבו החזיק כל גוף בהעתקי הגיבוי שלו במקום מרוחק באתר 
המחשבים שלו עצמו . הבעיה היותר מורכבת כאן , כמו גם במרכזי מחשוב אחרים , 
היא זו של מערך התקשורת . מי שעובדים בחיוג טלפוני רגיל יוכלו , כמובן , 
להתקשר גם אל מחשב הגיבוי . אולם הלקוחות הכבדים יותר של מרכזי המחשוב 
הממשלתיים נזקקים לקווי נל"ן ( קווים ישירים מנקודה לנקודה ) . כיום נעשה 
מאמץ מזורז להקמת מערכת שתאפשר למתג את הקווים המיוחדים האלה ממרכז מחשבים 
אחד למשנהו . אם לא יעלה המאמץ יפה , ייתכן שיהיה צורך לעבור מעבודה בקיוון 
ישיר לעבודת אצווה : עיבוד לאחר מעשה ולאחר הזנה ידנית של חומר באתר המחשבים . 
הלל דודאי , מנכ"ל המלם , מקווה כי המשבר הנוכחי יביא את האחראים לתקצוב 
מערכות המחשבים במגזר הציבורי לשיקול מחודש בסוגיית הקמתו של מרכז גיבוי 
ממשלתי . הגוף הציבורי המגובה ביותר במגזר האזרחי הוא חברת בזק . כאן נרכש 
רק באחרונה מחשב מרכזי חדש , המיועד לצורכי גיבוי בלבד . בבנקים הגדולים 
נראה כרגע מצבו של בנק דיסקונט משופר יחסית , ככל שהדברים נוגעים להיערכות 
לעבודה בשעת חירום . מחשבי הבנק פוצלו , והם מותקנים בשני בניינים שונים 
ומגבים זה את זה . לבנק הפועלים אתר מרכזי אחד , אולם בנק זה מתבסס גם 
על ציוד ניקסדורף , שיתרונו העיקרי ביזור מערכות בסניפים ובאזורים . תיאורטית 
יוכל הבנק להמשיך לתפקד , לפחות ברמת הסניף והאזור , וגם אז לתקופה מוגבלת 
אפילו תיפגע המערכת המרכזית שלו . בבנק לאומי , כך נדמה , מברכים כיום 
על הסכסוך הנצחי עם ועד העובדים , המונע זה שנים אחדות את העברת מרכז המחשבים 
של הבנק מלב תל אביב לקריית המחשבים החדשה , מתקן קשב בלוד . המחשבים המרכזיים 
בלוד , המעלים אבק וצוברים פחת בשאר ימי השנה מבלי להביא תועלת , הם כיום 
מרכז גיבוי מושלם גם אם אין מדובר בשלב הזה בגיבוי " חם " כמו זה של דיסקונט , 
שבו קיים העתק מושלם בזמן אמיתי של המערכת , אלא בגיבוי לעבודת אצווה בלבד . 
טוב פחות הוא מצבם של בנק המזרחי והבנק הבין - לאומי , שלמרכזי המחשב שלהם 
אין , לדעת מומחים , גיבוי של ממש . קפמאבינכמה וכמה גופים , שנתפסו בלתי 
מוכנים בכל הנוגע לגיבוי מחשביהם , עורכים בימים אלה מסע בדיקה מזורז בחיפוש 
אחר אופציות להשגת גיבוי מהיר . יונה שהם , מנכ"ל אוניקס טכנולוגיות ( 
לשעבר קונטרול דאטה ישראל ) , מציע זה זמן מה את השטח הפנוי , מערכות 
מיזוג האוויר , אספקת הכוח וקווי התקשורת הזמינים בידיו לשירותם של מי שזקוקים 
למרכז גיבוי כזה . לדברי שהם , ניתן להביא בתוך שבוע מחשב יבם 1803 משומש 
או מחשב תואם מתוצרת היטאצי ולהתקין במקום מערך גיבוי לכל דורש . כמו אברהם 
חסון מהיטאצי , מגלגל שהם את רעיון מרכז הגיבוי שלו זה זמן רב . אם תיגמר 
המלחמה מהר וללא פגיעות במערכי מחשוב חיוניים , הוא מעריך , תקטן הדרישה 
גם הפעם מבלי שישתפרו נוהלי הגיבוי הנהוגים אצלנו . מאיר פורמנסקי מיבם 
מעריך כי שיקום מתקן מחשב מרכזי , הכולל מערכות קירור , רצפה צפה ותקשורת , 
עשוי להימשך חודשים אחדים . בעיתות חירום , כאשר חלק מבעלי המקצוע הנדרשים 
בתהליך הבינוי והשיקום משרתים במילואים , הבעיה מחריפה עוד יותר . פורמנסקי 
מעריך , כי רק לארגונים מעטים בישראל יש הסכמי גיבוי העונים על צרכיהם . 
הוא מדגיש , כי מי שצריכים לקבוע את הצרכים האלה אינם אנשי יחידת המחשב , 
אלא המשתמשים עצמם , בדרג ההנהלה הבכיר ביותר . חשוב במיוחד , אומר פורמנסקי , 
לתרגל את השימוש במערך הגיבוי . רק תרגילים תכופים כאלה יכולים לחשוף את 
הליקויים הקיימים בתוכניות התיאורטיות . הצרה היא , שבלא מעט מקרים נעשה 
התרגול רק כדי לצאת ידי חובה . מי שמיהרו להבטיח את עצמם יותר מאחרים הם 
דווקא בתי התוכנה . כאן ברור למנהלים כי הנכס היחיד שבידיהם הוא התוכנה 
שפיתחו והם נוהגים בהתאם . שי סולה , מ"אומני תוכנה " , יצרנית מערכת 
ניהול הנתונים ומחולל היישומים ספיינס , המיועד למחשבים מרכזיים גדולים 
של יבם , מספר כי גרסאות קוד המקור של התוכנה פוזרו בכמה אתרים של לקוחות 
ברחבי המדינה , כמו גם בכספת נעולה באחד הבנקים . חברה שווייצית , שנכנסה 
לשותפות ב"אומני תוכנה " , העריכה את שווי התוכנה בששה מיליון דולר . יחד 
עם התוכנה נשמרות גם הבדיקות הממוחשבות המשמשות במהלך הפיתוח . הסכמים עם 
לקוחות בארץ ובחו"ל יאפשרו לבית התוכנה שימוש במחשב חלופי לזה שעליו הוא 
מבצע את הפיתוח אם ייפגע זה האחרון . אלי אלרואי , מנכ"ל בית התוכנה יבס , 
שומר גם הוא על קוד המקור של תוכנה , פרי פיתוחם של אנשיו באתרי הלקוחות , 
ומסביר כי הבעיה בהשגת מערכות בינוניות חלופיות מדגם אי - אס 004 , שעליו 
נעשה עיקר עבודת בית התוכנה שלו , אינה גדולה : זמני האספקה של הציוד הזה 
קצרים ויש מתקנים בארץ שעמם ניתן להגיע להסכמי גיבוי . התעוררות המודעות 
לצורך בגיבוי גררה שגשוג במכירת מדיה מגנטית סרטים וקלטות . ד"ר דוד מיטווך , 
מנכ"ל חברת י . א . מיטווך ובניו , המייצגת בארץ את יצרנית הסרטים והדיסקים 
3 , מדווח על הטסה בהולה ארצה מדיה מגנטית . הביקושים בשוק המקומי התרחבו 
באופן מהותי והיצרן האמריקאי נתן עדיפות להזמנות מישראל . תחום נוסף שבו 
ניכרה התעוררות במכירות הוא זה של הכספות חסינות האש , המיועדות לשמירת 
סרטים ודיסקים .
