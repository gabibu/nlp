בתגובה להכנסת הצבא הסובייטי לאזרבייגאן הושמעו מבירתה , באקו , שני איומים 
אולטימטיוויים שתוקפם עומד לפוג הלילה : האחד מפי החזית העממית של אזרבייגאן 
לפוצץ את מפעלי תעשיית הנפט , שפירושו , מלבד הנזק הכלכלי , תוספת של 
שואה אקולוגית באזור שכבר מוכה במפגעים סביבתיים חמורים , המקצרים את תוחלת 
החיים של תושביו . השני הצהרתו של הסובייט העליון המקומי , שכבר הודיע 
חד - צדדית על ביטול משטר החירום שהוטל בפקודת מוסקווה , כי בכוונתו להכריז 
על עצמאות הרפובליקה אם הצבא הסובייטי לא יסוג לאלתר . האיום בפרישה מבריה"ם 
מעורר מספר שאלות : האם ההנהגה המקומית ( שבעקבות הכנסת הצבא לבאקו פרשה 
מהמפלגה הקומוניסטית הסובייטית הלכה למעשה ) יחד עם החזית העממית , יכולות 
ומסוגלות להוות כוח פוליטי אלטרנטיווי לשלטונה של מוסקווה ? האם יש לשני 
הגופים הללו מושג ברור למה עלולה להתפתח הכרזת הפרישה , מעבר לליבוי התלהבות 
ההמונים הנכונים להפגין ברחובות ואף לעצור את הטנקים בגופם , ומה הן הדרכים 
המעשיות לביצוע הפרישה המוצהרת ? מבחינת עמדותיה של מוסקווה קל לשער שאין 
היא יכולה להרשות לעצמה להתקפל בפני איומי האזרים ולהוציא את כוחותיה מן 
האזור , ולו עקב החשש שהמדורה הבוערת בבאקו עשויה לפזר את ניצוצותיה לכל 
עבר , ולהצית אש גדולה ברחבי המדינה כולה , הסובלת מחוסר יציבות חברתית . 
הפתרון של החלפת הצבא הסובייטי בכוח שיטור של האו"ם , רעיון המוצע על ידי 
חוגים מסוימים הן בבאקו והן במוסקווה , יש בו כדי לפגוע בתדמיתה העצמית 
של בריה"ם כמעצמת - על ומדינה ריבונית מאוחדת . הגורמים המשפיעים על המצב 
הנוצר הם כה רבים ועצמאיים , והסכנה שתקרית מקומית תצית מחדש את כל הזירה 
היא כה מוחשית , עד כי הן ההנהגה המקומית באזרבייגאן והן ההנהגה המרכזית 
במוסקווה אינן יכולות לחזות את ההתפתחויות האפשריות . אך בין המגמות המסתמנות , 
אפשר כבר לבודד את זו הנוגעת לגורלו האישי של מיכאיל גורבצוב . נפל בחלקו 
להיות הדמות הטראגית של שלב המעבר בהיסטוריה של המדינה בין אם תמשיך להיקרא 
בריה"ם ובין אם לאו . בין אם תעבור הפיכת חצר ימנית או שמאלית , עם גורבצוב 
ליד הגה השלטון או בלעדיו . המאורעות המסתחררים לאחרונה באימפריה הסובייטית 
כבר מעידים בבירור על סוף עידן היסטורי העובר כעת על המדינה , ומצביעים 
כבר עתה על כך שהפרידה מהמשטר הפוליטי הקיים טומנת בחובה לא פחות אימה ושפיכות 
דמים מאשר הולדתו .
