באגף למיילדות וגינקולוגיה במרכז הרפואי " שבי ציון " ( רוטשילד ) בחיפה , 
בשיתוף עם המחלקה לחשמל של הטכניון , פותח מוניטור אלקטרוני לבדיקת פאראמטרים 
של סבל עוברי לפני הלידה . מנהל האגף , פרוף מרדכי שרף , הסביר כי " במצוקה 
עוברית טרם הלידה יש אחוז גבוה של טעויות ובשל כך יש אחוז גבוה של ניתוחים 
קיסריים מיותרים , עם כל הסיכון הכרוך בכך . המכשיר שפותח שהוא מוניטור , 
שבאמצעותו ניתן לבדוק פאראמטרים שונים של מצוקה . המכשיר נוסה במשך שלוש 
שנים על 003 מקרים חשודים ו006 מקרים נורמאליים . נמצאה התאמה בין תוצאות 
בדיקת המכשיר ומצב העובר . במהלך תקופה זו הצלחנו להוריד ב%03 את מספר הניתוחים 
הקיסריים . כיום נעשה ניסיון לקבל אישור של מנהל התרופות והמזון בארצות 
הברית , כדי להביא לייצור מסחרי ושיווק . ד"ר ראובן לוינסקי , שהיה שותף 
לפיתוח המכשיר , עוסק כיום בקנדה בניסוי המכשיר בחיות כדי להוכיח את אמינותו 
לשלטונות בארצות הברית " . בכנס הגינקולוגים האחרון , שנערך לפני מלחמת המפרץ , 
הציגו רופאי האגף גם פיתוח תבחין חדש לגילוי מוקדם של סיבוכים הריוניים , 
כמו רעלת היריון , פיגור תוך - רחמי , לידה מוקדמת ספונטאנית ואחרים , 
עוד בטרם הופעת הסימנים הקליניים . פרוף שרף ציין כי הפיתוח נעשה בשיתוף 
פעולה עם פרוף מיכאל זילברמן מהמחלקה ללימודים מורפולוגיים בפקולטה לרפואה 
של הטכניון . מדובר בבדיקת דם המגלה רמות של חלבון שלייתי ( פי - פי 31 ) 
. המחקר שהביא לפיתוח התבחין הקיף 649 נשים , שהמעקב אחר הריונן התבצע 
בתחנות טיפת חלב ובמרפאות סיכוני הריון , ומהן נלקחו 502 , 2 בדיקות דם . 
קבוצת המדגם הסופית כללה 663 נשים שהריונן הוגדר כתקין , וכן 83 נשים שסבלו 
מרעלת הריון , 05 שילדו ספונטאנית לפני השבוע ה73 להריון ו52 שילדו תאומים . 
המחקר הוכיח כי בעזרת התבחין החדש ניתן לאבחן ביותר מ%09 הצלחה את הסיבוכים , 
שבועיים ויותר לפני האבחנה הקלינית . פרוף שרף הוסיף כי מאחר שמדובר בבדיקת 
דם פשוטה לביצוע וזולה יחסית , יכול תבחין זה לשמש לסריקת אוכלוסיית הנשים 
בהריון . הוא הוסיף כי מתנהל מו"ם עם נציגות של חברה אמריקאית בארץ לפיתוח 
ערכה לגילוי החלבון , לצורך שימוש נרחב בתבחין .
