בניסיון של הרגע האחרון להנצל מהכיתור הגדול ומהתמוטטותן של 81 דיוויוזיות 
החי"ר בקו הראשון , הורה סדאם חוסיין לכוחותיו לסגת מכוויית . הסכנה היא 
שבדרך זו יציל השליט העיראקי את מכונת הלחימה שלו . ארה"ב ניצבת כרגע בפני 
בעיה קשה : האם לדחות את הצעת " התבוסה העיראקית " ולהכריז כי מטרות המלחמה 
הן רחבות בהרבה ממה שאישרה בשעתו מועצת הביטחון , דהיינו : הפלת משטרו 
של סדאם חוסיין . את זה יכולה ארה"ב לעשות בדרך אחת : לפרק את הצבא העיראקי 
מנשקו ולא לאפשר לו לסגת מכוויית עם כלי הנשק שלו , ובפרט הנשק הכבד . אם 
ידחו האמריקאים את הצעת הנסיגה מכוויית וימשיכו להילחם , רק מפני שעיראק 
טוענת כי מדובר בקבלת היוזמה הסובייטית , הם עלולים להיקלע לעימות עם בריה"ם 
במועצת הביטחון וייתכן שגם עם הצרפתים , שנשיאם הכריז לאחרונה כי המטרה 
איננה הפלתו של סדאם חוסיין . הדרך , אם כן , היא תבוסת הצבא העיראקי 
באמצעות נסיגה ללא נשק , בהנחה שכך יופל סדאם בארצו בשלב מאוחר יותר . קפידבינכוחות 
בעלות הברית תפסו ביומיים הראשונים של קרבות הקרקע תחמושת כימית רבה . מלבד 
פגזי ארטילריה ופגזי מרגמות כימיים , נתגלו לראשונה גם מוקשים כימיים במאגרים . 
הפגיעה בדיוויזיות החי"ר העיראקיות חשובה , משום שאילו היו חיילים אלה מתעקשים 
להחזיק בעמדותיהם , היו מסבים ודאי אבדות רבות לבעלות הברית . הרושם הוא 
שרצון הלחימה שלהם אינו גבוה . לא מן הנמנע , שרבים מהם יודעים כי שלטונות 
בגדאד היו מוכנים לוותר על כוויית במשא - ומתן , ושאלו את עצמם מדוע ליהרג 
במלחמה על מה שכבר ויתרו עליו מנהיגיהם . הפיקוד העיראקי לא הסתפק בהעברת 
הדיוויזיות הממוכנות והמשורינות לקו השני והשלישי , אלא הסיג גם את מרבית 
גדודי הטנקים , המשולבים בדיוויזיות החי"ר , שחלק מהם נתקל בכוחות בעלות 
הברית . מהלך זה נועד , כנראה , להציל את הטנקים כדי להכינם למתקפות הנגד . 
קפידבינבמהלך האוגף ממערב לכוויית , בתוך שטח עיראק , ביצעו אתמול בעלות 
הברית הנחתת ציוד וכוחות מהאוויר ליד העיירה סלמאן . מכאן ניתן ללמוד כי 
האיגוף הוא נרחב ביותר , וכי הכוח המאגף המערבי פונה לכיוון העיר סמאוה , 
השוכנת על גדת נהר הפרת בדרך הראשית בגדאד - בצרה . בדרך זו מועברת אספקה 
רבה לכוחות העיראקיים בכוויית , והכוונה היא לחסום אותה ולמנוע את מעבר 
האספקה . המאמץ השני הוא של הקורפוס המשוריין השביעי , המתקדם לאורך גבול 
עיראק - כוויית . כוחות מוטסים הונחתו בציר זה סמוך לעיירה אל - בוסייה , 
והקורפוס המשוריין חבר אליהם . כוח שריון עיראקי ירד לעברם . לפי אחת 
הידיעות , שייך כוח זה לדיוויזיה הממוכנת של משמר הרפובליקה . כאן התנהל 
קרב השריון הממשי היחיד ביזמת העיראקים , שלדברי בעלות הברית הוכו .
