ביקורו של שר החוץ הסובייטי אדוארד שוורדנדזה במטה הברית הצפון אטלנטית , 
הבהיר גם לאחרוני הספקנים כאן כי נאט"ו צועדת לקראת עידן חדש , שבו היא 
תיאלץ להגדיר מחדש את משימותיה , או אף לחשוב על אפשרות של פירוק . אחד 
מראשי הספקנים , המזכיר הכללי של נאט"ו מנפרד וורנר , הודה בימים אלה שהברית 
אכן תצטרך מעתה להתרכז בתפקידים חדשים . בצד הצורך להתמסר גם להבא למשימה 
של " מניעת מלחמה על ידי שמירת כושר ההגנה " , מנה וורנר פיקוח על פירוק 
הנשק , ו"הקמת סדר אירופי חדש , מבוסס על יציבות , חופש ושלום " . בסדר 
כזה צריכה להשתלב גם ארה"ב , שכוחותיה יידרשו גם בעתיד לקיים כאן נוכחות 
מסוימת , ולתת בכך תוקף ל"סולידריות האטלנטית " של 61 המדינות החברות בנאט"ו . 
ברוח דומה , אם כי בהדגש שונה , התבטא השבוע גם שר החוץ המזרח גרמני אוסקר 
פישר . בשידור בשירות הבין - לאומי של רדיו מזרח ברלין אמר פישר , כי שתי 
בריתות ההגנה היריבות באירופה עומדות בפני המשימה של צמצום תפקידיהן הצבאיים 
והעמקת מעורבותן בתהליכים הפוליטיים העוברים על היבשת . לדעת שר החוץ המזרח 
גרמני , התהליך הזה , שכבר התחיל להתרחש בשטח , יוביל במוקדם או במאוחר 
להיעלמות הדרגתית של שתי הבריתות . החזון המקופל בדברי פישר פירוק הדדי 
של נאט"ו ושל ברית וארשה מקובל כיום מאוד על המזרח אירופאים , שרוצים להיפטר 
בהדרגה מן הנוכחות הצבאית הסובייטית על אדמתם ( במזרח גרמניה בלבד חונים 
קרוב ל004 אלף חיילים של הצבא האדום ) . אבל גם במערב אירופה רואים יותר 
ויותר כריאליסטית וכרצויה את האפשרות לקיצוצים מרחיקי לכת בכוחות של שתי 
מעצמות העל . השאלה המוצגת בדעת הקהל כאן , והמהדהדת גם במטה נאט"ו עצמו , 
היא בשביל מה להחזיק בעוצמה צבאית כה גדולה כאשר האימפריה הסובייטית מתפרקת 
מאליה , והסבירות של מלחמה חדשה נראית רחוקה יותר מאשר אי פעם מאז סוף שנות 
הארבעים . תשובת הקסם הניתנת כאן , שבאה לידי ביטוי גם בדברי המזכיר הכללי , 
היא " סכנת אי - היציבות " . נשמעת הטענה המוצדקת , שתהליכי הרפורמה במזרח 
אירופה טרם הבשילו , ושגם בברית המועצות עצמה עדיין מוטל בספק גורלן של 
הפרסטרויקה והגלאסנוסט , ועמן גם עתידו של הנשיא מיכאיל גורבצוב . עדיין 
ייתכן מהפך , צבאי או אחר , שיעלה את הסובייטים מחדש על קו של עוינות ותוקפנות . 
יתר על כן , המתכננים הצבאיים אינם יכולים לבסס את שיקוליהם על כוונות 
יפות , אלא על עוצמה ממשית בשטח . ובתחום הזה ( וכנראה בו בלבד ) , 
היתה בריה"ם ונשארה מעצמת על בעלת כושר התקפי המחייב תשובה מתאימה . נימוקים 
כאלה , ככל שהם תקפים , מאבדים ממשקלם נוכח הסיכויים להשיג פריצת דרך דרמאטית 
בקיצוץ הכוחות הקונוונציונליים של שתי מעצמות - העל ( ואולי לא רק שלהן ) 
בחזית האירופית . מומחה ההגנה של המפלגה הדמוקרטית האמריקאית , הסנטור 
סאם נאן , הגדיר בימים אלה כמיושנות את ההצעות של הנשיא גורג בוש מתחילת 
השנה שעברה , לקצץ את כוחותיה של ארה"ב ביבשת ל572 אלף חיילים . לדעת 
נאן ייתכן קיצוץ הדדי עד לתקרה של 002 אלף חיילים של כל אחת משתי מעצמות - 
העל . אחד הנימוקים המעניינים שהעלה נאן : נוכחות צבאית סובייטית ואמריקאית 
בהיקף שהציע בוש תיתן לגיטימציה להמשך ה"כיבוש " למעשה של מדינות מזרח אירופיות 
המשתחררות מעול הקומוניזם . מעבר לכך , נוכחות כזאת היא כנראה בהיקף גדול 
יותר מכפי שגורבצוב עצמו רוצה ומוכן לקבל לאור המצב החדש שהשתרר בפולין , 
הונגריה , צכוסלובקיה ומזרח גרמניה ( שבה נחשפים הכוחות הסובייטיים הגדולים 
ליותר ויותר גילויי עוינות מצד האוכלוסייה המקומית ) . ברחבי אירופה ( 
כולל מזרח היבשת ) מתגברים הלחצים על הממשלות לקצץ בכוחות המזוינים , ולהפנות 
את המשאבים המשתחררים למשימות אחרות . כמעט ודאי שגם המתכננים של נאט"ו 
יצטרכו להסתגל למצב חדש , שבו " מצבת כוח האדם " מתקצצת מאליה , בגלל 
העדר יכולת של הממשלות הבודדות לעמוד בהתחייבויותיהן כלפי הברית . באיחור 
מסוים מסתגלים גם כאן לעובדות החיים החדשות . העימות הצבאי במרכז אירופה 
כמעט שלא קיים עוד בתודעה הציבורית , ובמשך הזמן הוא יישחק בוודאי גם 
במונחים של מספרי גייסות , קני ארטילריה וטילים גרעיניים . המשימה החדשה 
של " ארגון הסדר האירופי לשנות התשעים " מצטיירת כיום כחשובה יותר מהגברת 
העוצמה הצבאית . שתי הבריתות הצבאיות באירופה , חוזה אחד הדיפלומטים כאן , 
יהיו עסוקות בעתיד בבלימת הדהירה לקראת האיחוד הבלתי נמנע של שתי הגרמניות , 
ובגיבוש הצעות למערכת של הסדרי שלום בגבולותיה של המדינה המורחבת . הוויכוחים 
הסוערים על טילים גרעיניים נוספים לטווחים קצרים , שאיימו להטיל את נאט"ו 
למשבר קיומי עוד באביב של שנת 9891 , מוזכרים כאן רק כהד קלוש לעידן שחלף 
ואיננו עוד .
