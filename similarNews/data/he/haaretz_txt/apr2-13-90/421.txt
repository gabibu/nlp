יו"ר איגוד הכדוריד הישראלי מני וייצמן שיגר מברק ברכה חם ליו"ר איגוד הכדוריד 
של נורווגיה . במברקו מברך וייצמן את היו"ר הנורווגי לרגל ההצלחה הגדולה 
של נורווגיה שתפסה את המקום הראשון בבית המוקדם באליפות העולם דרג ב הנערכת 
בפינלנד , ובכך הבטיחה את עליתה לדרג ב העולמי . אבל וייצמן לא הסתפק בברכה 
שיגרתית . מאחר שישראל תשחק מחר נגד מערב גרמניה החזקה , עם סיכויים מעטים 
לניצחון , חישב וייצמן ומצא , כי אם הנורווגים יזלזלו מחר במשחקם עם הבלגים , 
יכולים הבלגים לנצח את נורווגיה ולעלות למקום השלישי בטבלה , כדי להותיר 
את ישראל בידיים ריקות ולהבטיח לבלגים את העליה לדרג ב . היו"ר מני וייצמן 
מיהר איפוא לבקש מעמיתו הנורווגי לא לזלזל בבלגים ולהשקיע גם מולם , למרות 
שאין המשחק הזה משנה למעמד הנורווגים שכבר הבטיחו לעצמם את הבכורה , את 
כל כוחם .
