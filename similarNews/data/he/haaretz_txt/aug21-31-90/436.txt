במסגרת המאמצים להכין את העורף למלחמה הבאה מתקיימים ימי הדרכה בנושא אב"ך 
בשיתוף של צה"ל , הג"א והרשויות המקומיות . תוכנית ההדרכה פועלת מזה כשנה 
וחצי בתדירות של מפגש לחודשיים - שלושה , אולם מאז הפלישה העיראקית לכוויית 
ניכרת היענות גדולה ביותר לנושא , ורשויות מקומיות החלו להתאים עצמן לדרישה . 
ברמת - השרון התקיימו כבר שלושה ימי הדרכה מאז פלישתה של עיראק , והשתתפו 
בהם בין 000 , 2 ל000 , 3 תושבים , על - פי הערכת המארגנים . במפגש שהתקיים 
אתמול ברמת - השרון הסבירו לקהל על גז חרדל , גז עצבים , כיצד ניתן להתגונן 
ולהגן על בני המשפחה בעת התקפת גזים , כיצד ניתן לאטום חדר וכיצד להכין 
את המקלט . הודגם השימוש במסכות אב"ך , בברדס המיועד לילדים בני 03 ובעריסה 
מיוחדת לתינוקות , תוך הסתייעות בציוד המיגון ובסרטון וידאו . כן חולקה 
לבאים רשימה של נקודות לחלוקת מסכות אב"ך ברמת - השרון . לאחר ההסבר עבר 
הקהל לחדר שבו יכלו למדוד את המסכות ולקבל הדרכה אישית לגבי השימוש בציוד . 
שוב ושוב נשמעו טרוניות כמו " זה נורא לא נוח " , " אי אפשר לנשום דרך 
זה " . קלריס דיין , מנהלת שירותי החירום במועצה המקומית , סיפרה כי תושבי 
רמת - השרון המגיעים להדרכה אינם שרויים בפאניקה . " אני רואה אנשים מודעים 
שבאו ללמוד " , היא מעידה . " הם שואלים שאלות כגון כמה זמן מראש תינתן 
ההתראה , באיזו צורה , מתי נקבל מסכות , האם יהיה לנו מספיק זמן להתגונן " . 
גם מדריך צה"ל העיד כי התושבים מתייחסים לנושא בצורה נכונה . " שואלים 
שאלות ענייניות כמו כמה שעות תימשך ההתקפה , איזה אמצעי מיגון יש , הסיכויים 
לפרוץ התקפה , מה היקף האוכלוסייה שעלולה להיפגע " . למפגש שהתקיים אתמול 
בשעות אחר - הצהרים הגיעו רק נשים וילדים . אריאלה שלם , למשל , הגיעה 
בגפה . " הבעל והבן הגדול שלחו אותי . בני בן השמונה לא רצה לבוא . מפחד . 
קיבלתי קצת מושג . עכשיו אני יותר חרדה , כי אני יודעת מה זה " . אילן 
קריין , מתנדב צעיר מהג"א , אמר כי בשעות הערב מגיעים גם הגברים , אולם 
בדרך כלל נשים הן הדומינאנטיות בקהל . לדעתו הדבר נובע מכך שגברים מקבלים 
הדרכה בצבא על השפעת הגזים וכיצד ניתן להתגונן . קפידבינבין התשובות שניתנו 
לשאלות תושבים : לבעלי - חיים אין שום הגנה בעת התקפה כימית ; רצוי להשתמש 
במזון אטום ; חולי אסתמה יכולים לקבל ברדס במקום מסכה , שבאמצעותו הנשימה 
חופשית יותר ; משקפיים ועדשות - מגע יש להסיר לפני שעוטים את המסכה . בצה"ל 
מסבירים כי תוכנית ההדרכה להכנת העורף , הן בתחום הקונבנציונאלי והן בתחום 
הכימי , מופעלת ללא קשר לפלישה העיראקית לכוויית . במסגרת התוכנית פותח 
מקצוע חדש לחיילות צה"ל , מדריכות אלש"ח ( אוכלוסייה לשעת חירום ) . 
כ03 חיילות הוכשרו לתפקיד זה בחצי השנה האחרונה .
