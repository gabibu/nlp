רה נדירה , המיוחסת לתקופה הרומית , התגלתה באחרונה ליד קיבוץ שניר שבגולן . 
המערה נגלתה לאחר שפרה החליקה בפתחו של בור . כשהחלו לחלצה הבחינו בין מפולות 
העפר בממצאים ארכיאולוגיים בלתי מזוהים . מחלצי הפרה הזעיקו למקום פקח של 
רשות העתיקות , ובבדיקת הממצאים התברר שמדובר במערה עתיקה , כנראה מהתקופה 
הרומית . נמסר , כי באותו אזור נחשף בעבר אתר בורות קבורה עתיקים , אולם 
עד כה לא היה ידוע על קיומו של הבור הנוכחי . בעקבות הגילוי החדש , מתגבשת 
תוכנית לביצוע סדרת חפירות חדשה , במטרה לחשוף את כל בורות הקבורה המצויים 
באזור .
