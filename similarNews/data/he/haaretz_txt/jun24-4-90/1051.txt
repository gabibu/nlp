יו"ר מפלגת העבודה , ח"ך שמעון פרס , יחליט רק הבוקר אם יופיע אישית בפני 
ועדת החוקה של מפלגת העבודה , המתכנסת בצהריים לשמוע את טענותיו נגד דרישתו 
של יצחק רבין לקיום התמודדות על הנהגת המפלגה . בסוף השבוע סיכם פרס עם 
עו"ד שלום פדידה , ראש עיריית בית שמש , שהוא ייצגו בוועדה . פדידה היה 
מתומכי פרס בוועדת החקירה המפלגתית בראשותו של פרוף יורם לס לבירור סיבות 
כישלונה של מפלגת העבודה בבחירות . ועדת החוקה של מפלגת העבודה , בראשותו 
של ח"ך דוד ליבאי , תדון היום , לאחר שתשמע את טענותיו של פרס או של בא 
כוחו , ביותר משלושים שאלות שהפנה אל הוועדה מזכ"ל מפלגת העבודה , ח"ך 
מיכה חריש . בין השאר שואל חריש האם המהלך של רבין הוא אכן הדחה או התמודדות 
לגיטימית , והאם פרס רשאי לדרוש הצבעת אמון במרכז המפלגה לפני כל שאלה בנושא 
ההתמודדות . הערב , או לכל היותר מחר , תמסור הוועדה את מסקנותיה למזכ"ל 
המפלגה , ולפי החלטתה תכונס מחדש לשכת מפלגת העבודה ביום רביעי או חמישי 
להכריע בשאלת ההתמודדות . קפידבינתומכי פרס במפלגת העבודה האיצו בסוף השבוע 
את מאמציהם להחתים חברי מרכז על הצבעת אמון בפרס . בכוונתם לנסות ולמנוע 
את כינוס לשכת המפלגה , מחשש לנחול הפסד בה , ולהעדיף כינוס ישיר של מרכז 
המפלגה , אשר בו יבקש פרס אמון מחברי המפלגה . מטה ח"ך יצחק רבין התכנס 
אתמול ויתכנס גם הבוקר לדון בניסיונותיהם של פרס ואנשיו לכנס את מרכז המפלגה 
ולהצביע בו אמון . אנשי רבין מבקשים להיאחז בתקנון המפלגה , הקובע כי דרישה 
לאמון מחודש פירושה התמודדות , ובמהלך שכזה יכולים להשתתף כמה מתמודדים . 
אנשי רבין , הממתינים להחלטת ועדת החוקה , יתחילו מחר באיסוף חתימותיהם 
של חברי מרכז , לכינוס דחוף של המרכז שבו יוחלט על התמודדות מידית . קפידבינבעקבות 
מינויו של ח"ך מוטה גור לראש מטה הפעילות של יצחק רבין , החליט שמעון פרס 
למנות את ניסים זווילי לראש מטהו . הערב יתכנס מטה פרס בראשונה , וככל הנראה 
יסוכם סופית אם על פרס לדרוש הצבעת אמון בכינוס המרכז . ח"ך אברהם בורג , 
שהיה בין שישה חברי הכנסת שיעצו בשבוע שעבר לפרס לוותר על ההתמודדות מחשש 
התבזות , התקשר בערב השבת לפרס , ומסר לו כי החליט לתמוך בו , מאחר שהוא 
מתנגד להדחה ועריפת ראשים . בורג הבהיר , כי אם יחליטו חבריו להציג מועמד 
שלישי , בן הדור הצעיר , הוא עשוי לתמוך במועמד הצעיר . קפידבינחברי נוער 
עבודה , שלהם זכות השתתפות פאסיווית בלשכת מפלגת העבודה , הודיעו אתמול 
כי כצעד הפגנתי יציגו גם הם מועמד משלהם בהתמודדות בין רבין ופרס להנהגת 
המפלגה . ח"ך גד יעקבי אמר אמש כי ברור לו שהמהלך של יצחק רבין לשמש כראש 
האופוזיציה במקומו של פרס , אין פירושו של דבר בחירת מועמד המפלגה לראשות 
הרשימה לקראת הבחירות ב2991 , וכי בדעתו להתמודד על ראשות מפלגת העבודה 
לקראת הבחירות שיתקיימו ב2991 .
