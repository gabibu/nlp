לפני שבועות מספר הוקמה במועצת העיתונות ועדת בירור שדנה בתלונה שהגישה נגדי 
חברת נירוסטה בע"ם . הרקע לתלונה הוא ידיעה שפירסמתי ב"הארץ " , שבה דווח 
כי בנק שווייצי ( " סוויס - בנק " ) הגיש לבית המשפט המחוזי בתל - אביב 
בקשה לפרק את החברה הישראלית ( נירוסטה ) בגין חוב . ועדת הבירור קבעה 
במסקנותיה , כי שגיתי בכך שלא פירסמתי ידיעה מלאה הנוגעת לבקשת הפירוק . 
הוועדה כתבה כי הידיעה " החסירה את החלטתו של השופט שדחה את בקשת הפירוק " . 
מסקנותיה של ועדת הבירור של מועצת העיתונות פורסמו באחרונה , לרבות בעיתון 
" הארץ " . אינני יכול להתייחס ברצינות הראויה לממצאי הוועדה , כי אותה 
ועדה החליטה שלא להתייחס במסקנותיה לראיות אובייקטיוויות שנמסרו לה מתוך 
תיק המשפט . זאת ועוד : כל הקורא את ממצאי הוועדה עלול לקבל תמונה מעוותת 
באשר לגורל בקשת הפירוק שהוגשה נגד החברה . כדי לדעת במה דברים אמורים , 
מן הראוי לציין את העובדות הבאות : יום אחרי שהוגשה בקשת הפירוק דן בה 
נשיא בית המשפט המחוזי בתל - אביב , ד"ר אליהו וינוגרד . השופט וינוגרד 
ציין בתחתית הבקשה ( 98 . 7 . 7 ) , כי " הבקשה אינה נתמכת בתצהיר " . 
כמו - כן כתב השופט כי חשש בלבד אינו יכול לשמש בסיס לפירוק זמני . אם 
ישנן עובדות מעבר לכך יש לפרטן " . אגב , לא מצאתי בהחלטתו של השופט כי 
דחה את בקשת הפירוק . ראוי להדגיש כי השופט וינוגרד , כעבור חמישה ימים 
( 7 . 21 ) , כתב החלטה חדשה שבאה לתקן למעשה את החלטתו הקודמת , כי 
לא הבחין ( שלא " באשמתו " , כדברי פרקליט הבנק ) כי לתיק בית המשפט צורף 
מלכתחילה תצהיר מטעם הבנק השווייצי . השופט הבחין בו באיחור אחרי שפרקליט 
מטעם הבנק עלה ללשכתו והסב את תשומת לבו . ואכן , חמישה ימים אחרי ההחלטה 
הראשונה של השופט , שבה כתב כי אין תצהיר מצורף , כתב החלטה נוספת בזו 
הלשון : " לזמן במעמד הצדדים אחרי הפגרה " . במלים אחרות : בקשת הפירוק 
הוגשה כדין וזו הסיבה שנקבע מועד לדיון בבקשת הבנק . שבעה ימים אחרי ההחלטה 
המתוקנת של השופט פנה פרקליט נירוסטה , עו"ד דניאל הופר , למועצת העיתונות 
וטען , כי הפרסום פגע בחברה , כי לא הוזכרה הערתו של השופט כי " הבקשה 
אינה נתמכת בתצהיר " . עו"ד הופר לא הזכיר , משום - מה , שקיימת כבר החלטה 
חדשה של השופט שקבעה את הדיון בבקשת הפירוק אחרי הפגרה . עו"ד הופר טען 
באוזני , כי לא ידע אז על החלטתו החדשה של השופט ( כי קיים תצהיר ) , 
למרות שחלפו כבר שבעה ימים מאז מתן ההחלטה . העתק ממכתבו של עו"ד הופר שוגר 
למערכת " הארץ " . אני , ביוזמתי , צילצלתי אליו ואמרתי לו שאני מוכן 
לתקן את הידיעה כבר למחרת ( עשרה ימים אחרי הפרסום ) . ראוי לציין , 
כי בשלב זה לא ידעתי כלל על ההחלטה החדשה של השופט וינוגרד המלמדת כי בקשת 
הפירוק הוגשה כראוי . הוזמנתי על - ידי חברי ועדת הבירור במועצת העיתונות . 
חברי הוועדה היו : גבריאל צפרוני ( יו"ר ) , אלתר ולנר וגד ליאור . 
לישיבת הבירור הבאתי את המסמכים מתיק בית המשפט . חזרתי והדגשתי באוזניהם , 
שיתעלמו מגרסתי - תגובתי בעל - פה , ושישימו לב רק למסמכים שאני מוסר להם . 
מסרתי לחברי הוועדה מסמכים , מתוך הנחה שהם יעיינו בהם ויתייחסו אליהם בממצאיהם . 
אלה הם שלושת המסמכים שנמסרו להם : א . תצהיר של מנהל הבנק השווייצי המלמד 
כי אכן הבקשה לפירוק החברה הוגשה כראוי . ב . החלטתו המתוקנת של השופט 
וינוגרד . ג . צילמתי את כריכת תיק בית המשפט והראיתי להם שבקשת הפירוק 
גם נכון להיום תלויה ועומדת ונקבעה ל12 בפברואר . ראוי עתה לשאול : מדוע 
לא כתבו חברי ועדת הבירור , שהבקשה עדיין תלויה ועומדת ? תיק בית המשפט 
מלמד למשל , כי עוד לפני הגשת בקשת הפירוק פנתה עיריית תל - אביב לחברת 
נירוסטה והזהירה אותה , כי אם לא תשלם את חובה לעירייה ( צקים של החברה 
לא כובדו על - ידי הבנק ) היא עלולה לבקש את פירוקה ( העירייה הצטרפה לבקשת 
הפירוק אחרי הגשת הבקשה על - ידי הבנק השווייצי ) . מה עלול הקורא להבין 
מקריאת מסקנות חברי ועדת הבירור במועצת העיתונות ? שפשוט , נכון להיום , 
נדחתה בקשת הפירוק במשמעות של בקשה ש"נמחקה " , ש"נעלמה " . וזאת , הואיל 
והוועדה כתבה כי הידיעה " החסירה את החלטתו של השופט שדחה את בקשת הפירוק " . 
הרי צירפתי את צילום כריכת תיק בית המשפט שבו רשום כי הדיון בבקשת הפירוק 
נקבע ל12 בפברואר . כיוון שכך , אפשר לומר שחברי ועדת הבירור של מועצת 
העיתונות הרחיקו - לכת כשכתבו דבר שגם השופט לא כתב בהחלטתו . ההחטאה הגדולה 
ביותר של ועדת הבירור היתה , כי הידיעה שכתבתי , הגם שהחסירה את הערתו 
של השופט באשר להעדרו של התצהיר , היתה עובדתית נכונה . הראיה לכך : התצהיר 
נמצא על - ידי השופט בתיק בית המשפט , והראיה גם שהוא שינה את החלטתו . 
במסקנות הוועדה אין זכר לאחד משלושת המסמכים שמסרתי לחבריה . למרבה הפלא 
אין לוועדת הבירור במועצת העיתונות ערכאת ערעור שקיימת בכל בית משפט . אפילו 
בבית המשפט העליון עצמו יש , במקרים נדירים , " מעין ערעור " , כשבית 
המשפט מתכנס לדיון נוסף בהרכב מורחב . אין זה מקובל שכתב עוסק בעניין פרטי . 
אך באין אפשרות לערער על החלטת מועצת העיתונות , אין לי ברירה אלא להביא 
לידיעת הציבור את העובדות .
